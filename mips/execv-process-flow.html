execv/execve：把当前进程变成另一个程序

用户常说“execv 创建进程”，但内核事实更精确：`execv()` 是 libc 的 exec 家族接口，最终落到 `execve(2)` 语义；成功时它不返回，也不分配新的 PID，而是在当前 task 上拆掉旧用户地址空间，装入新 ELF 或解释器脚本，重建用户栈，最后让 MIPS 的 EPC 指向新入口。

官方依据

主题官方链接本文采用的事实
exec 家族exec(3)`execv()` 等 libc 接口 layered on top of `execve(2)`；`v` 表示参数以向量数组传入；无 `e` 后缀时环境来自 `environ`。
execve 系统调用execve(2)`execve()` 替换当前进程映像；成功不返回；PID 不变；捕获的信号处理、内存映射、alt stack 等属性会被重置或丢弃。
ELF 格式elf(5)ELF 由 ELF header、program header 等组成；`PT_LOAD` 描述可加载段，`PT_INTERP` 描述动态链接器路径，`e_entry` 是开始执行的虚拟地址。
第一性原理概述：“进程”由 task 身份、资源引用和用户态程序映像共同构成。exec 改的是程序映像，不是 task 身份；因此 PID 保持，地址空间和入口点被替换。

fork + exec 才是“创建并运行新程序”

图 1：shell 常见流程是 fork 后子进程 exec；单独 exec 只是当前进程改头换面。
本文标题沿用“execv 创建进程”的常见说法，但正文按内核事实表述：`execv()` 不是创建新 task，而是让已有 task 执行新程序。
第一性原理概述：创建新调度实体需要复制或创建 `task_struct`；exec 路径没有做这件事，它是在当前 `current` 上安装新的 `mm_struct` 和寄存器返回点。

系统调用入口：`execv()` 进入内核 `execve`

/* fs/exec.c */
SYSCALL_DEFINE3(execve,
                const char __user *, filename,
                const char __user *const __user *, argv,
                const char __user *const __user *, envp)
{
        return do_execve(getname(filename), argv, envp);
}

SYSCALL_DEFINE5(execveat,
                int, fd, const char __user *, filename,
                const char __user *const __user *, argv,
                const char __user *const __user *, envp,
                int, flags)
{
        int lookup_flags = (flags & AT_EMPTY_PATH) ? LOOKUP_EMPTY : 0;

        return do_execveat(fd,
                           getname_flags(filename, lookup_flags, NULL),
                           argv, envp, flags);
}

逐段解释：用户态 `execv(path, argv)` 没有显式 `envp` 参数，glibc 会使用当前 `environ` 并调用内核 `execve` 语义。内核真正看到的是 `filename`、`argv`、`envp` 三组用户指针。

`getname()` 先把用户态路径名复制进内核。`execveat()` 多了 fd 和 flags，可支持相对 fd 或 `AT_EMPTY_PATH` 等形式；最终两者都进入 `do_execveat_common()`。

第一性原理概述：系统调用边界先把“不可信用户指针”变成“内核可验证对象”。路径名、参数数组和环境数组都必须经过长度、权限和访问检查。

`do_execveat_common()`：准备 linux_binprm 与临时参数栈

/* fs/exec.c */
static int do_execveat_common(int fd, struct filename *filename,
                              struct user_arg_ptr argv,
                              struct user_arg_ptr envp,
                              int flags)
{
        struct linux_binprm *bprm;
        int retval;

        if (IS_ERR(filename))
                return PTR_ERR(filename);

        if ((current->flags & PF_NPROC_EXCEEDED) &&
            atomic_read(&current_user()->processes) > rlimit(RLIMIT_NPROC)) {
                retval = -EAGAIN;
                goto out_ret;
        }

        current->flags &= ~PF_NPROC_EXCEEDED;

        bprm = alloc_bprm(fd, filename);
        if (IS_ERR(bprm)) {
                retval = PTR_ERR(bprm);
                goto out_ret;
        }

        retval = count(argv, MAX_ARG_STRINGS);
        ...
        bprm->argc = retval;

        retval = count(envp, MAX_ARG_STRINGS);
        ...
        bprm->envc = retval;

        retval = bprm_stack_limits(bprm);
        ...
        retval = copy_string_kernel(bprm->filename, bprm);
        bprm->exec = bprm->p;
        retval = copy_strings(bprm->envc, envp, bprm);
        retval = copy_strings(bprm->argc, argv, bprm);
        ...
        retval = bprm_execve(bprm, fd, filename, flags);
out_free:
        free_bprm(bprm);
out_ret:
        putname(filename);
        return retval;
}

逐段解释：`linux_binprm` 是 exec 的“施工图”：保存文件、解释器、凭据、参数数量、临时 mm、栈位置和二进制格式加载需要的信息。

代码先处理 `PF_NPROC_EXCEEDED`，这是官方 `execve(2)` 提到的 `EAGAIN` 来源。随后 `alloc_bprm()` 创建 `bprm` 并初始化一个尚未对用户可见的新 `mm`，再统计 `argv/envp`，计算栈限制，按倒序把文件名、环境和参数复制进临时参数页。

第一性原理概述：exec 不能一开始就破坏旧进程映像，因为后面可能因为权限、格式或内存失败而返回错误。内核先在 `bprm` 和临时 mm 中预构造新世界，直到确认可提交。

参数与环境变量：为什么倒序复制

/* fs/exec.c */
static int bprm_stack_limits(struct linux_binprm *bprm)
{
        unsigned long limit, ptr_size;

        limit = _STK_LIM / 4 * 3;
        limit = min(limit, bprm->rlim_stack.rlim_cur / 4);
        limit = max_t(unsigned long, limit, ARG_MAX);

        ptr_size = (max(bprm->argc, 1) + bprm->envc) * sizeof(void *);
        if (limit <= ptr_size)
                return -E2BIG;
        limit -= ptr_size;

        bprm->argmin = bprm->p - limit;
        return 0;
}

static int copy_strings(int argc, struct user_arg_ptr argv,
                        struct linux_binprm *bprm)
{
        while (argc-- > 0) {
                const char __user *str;
                int len;
                unsigned long pos;

                str = get_user_arg_ptr(argv, argc);
                len = strnlen_user(str, MAX_ARG_STRLEN);
                if (!valid_arg_len(bprm, len))
                        goto out;

                pos = bprm->p;
                str += len;
                bprm->p -= len;
                if (bprm->p < bprm->argmin)
                        goto out;
                ...
                if (copy_from_user(kaddr+offset, str, bytes_to_copy))
                        goto out;
        }
        return 0;
}

`bprm_stack_limits()` 把参数和环境字符串限制在栈上限的 1/4、`_STK_LIM` 的 3/4 与历史 `ARG_MAX` 规则之间，并预留 `argv[]/envp[]` 指针数组空间。超限返回 `-E2BIG`。

`copy_strings()` 从最后一个字符串开始倒序复制，因为最终用户栈从高地址向低地址增长。字符串先放到低处，稍后 `create_elf_tables()` 再把指针数组、`argc` 和 auxv 放到栈顶布局中。

第一性原理概述：新程序入口需要看到标准 C ABI 的栈布局，但内核拿到的是旧地址空间中的用户指针数组。exec 的参数复制就是把旧世界里的字符串冻结到新世界的初始栈里。

`bprm_execve()`：打开文件、准备凭据、进入二进制加载

/* fs/exec.c */
static int bprm_execve(struct linux_binprm *bprm,
                       int fd, struct filename *filename, int flags)
{
        struct file *file;
        struct files_struct *displaced;
        int retval;

        io_uring_task_cancel();

        retval = unshare_files(&displaced);
        if (retval)
                return retval;

        retval = prepare_bprm_creds(bprm);
        if (retval)
                goto out_files;

        check_unsafe_exec(bprm);
        current->in_execve = 1;

        file = do_open_execat(fd, filename, flags);
        retval = PTR_ERR(file);
        if (IS_ERR(file))
                goto out_unmark;

        sched_exec();

        bprm->file = file;
        ...
        retval = security_bprm_creds_for_exec(bprm);
        if (retval)
                goto out;

        retval = exec_binprm(bprm);
        if (retval < 0)
                goto out;

        current->fs->in_exec = 0;
        current->in_execve = 0;
        rseq_execve(current);
        acct_update_integrals(current);
        task_numa_free(current, false);
        if (displaced)
                put_files_struct(displaced);
        return retval;
        ...
}

逐段解释：`io_uring_task_cancel()` 取消跨 exec 不应保留的 io_uring 活动。`unshare_files()` 让文件描述符表私有化，方便安全地处理 close-on-exec。

`prepare_bprm_creds()` 准备新凭据并持有 `cred_guard_mutex`；`check_unsafe_exec()` 标记 ptrace、no_new_privs、共享 fs 等不安全条件；`do_open_execat()` 以执行语义打开目标文件，后续 LSM 可基于文件和凭据做检查。

第一性原理概述：exec 是安全边界：它可能提升权限、关闭 fd、改变 dumpable、重置信号处理。内核必须先把“可执行文件”和“将要生效的凭据”绑定起来，再允许二进制加载器提交。

static int exec_binprm(struct linux_binprm *bprm)
{
        int ret, depth;

        for (depth = 0;; depth++) {
                struct file *exec;
                if (depth > 5)
                        return -ELOOP;

                ret = search_binary_handler(bprm);
                if (ret < 0)
                        return ret;
                if (!bprm->interpreter)
                        break;

                exec = bprm->file;
                bprm->file = bprm->interpreter;
                bprm->interpreter = NULL;
                ...
        }
        audit_bprm(bprm);
        trace_sched_process_exec(current, old_pid, bprm);
        ptrace_event(PTRACE_EVENT_EXEC, old_vpid);
        proc_exec_connector(current);
        return 0;
}

`prepare_binprm()` 读取文件头前 `BINPRM_BUF_SIZE` 字节。每个注册的 `linux_binfmt` 尝试识别它：不认识返回 `-ENOEXEC`，认识就继续加载。

`exec_binprm()` 支持解释器重写，例如脚本把 `bprm->interpreter` 设置成解释器文件，循环再用解释器文件重新走 binfmt，最多允许有限层级，防止递归脚本无限套娃。

不可回退点：`begin_new_exec()` 拆旧世界、提交新凭据

/* fs/exec.c */
int begin_new_exec(struct linux_binprm * bprm)
{
        struct task_struct *me = current;
        int retval;

        retval = bprm_creds_from_file(bprm);
        if (retval)
                return retval;

        bprm->point_of_no_return = true;

        retval = de_thread(me);
        if (retval)
                goto out;

        set_mm_exe_file(bprm->mm, bprm->file);
        would_dump(bprm, bprm->file);

        acct_arg_size(bprm, 0);
        retval = exec_mmap(bprm->mm);
        if (retval)
                goto out;

        bprm->mm = NULL;
        ...
        do_close_on_exec(me->files);
        ...
        flush_signal_handlers(me, 0);

        security_bprm_committing_creds(bprm);
        commit_creds(bprm->cred);
        bprm->cred = NULL;
        ...
        security_bprm_committed_creds(bprm);
        return 0;
out:
        return retval;
}

`begin_new_exec()` 是执行路径的分水岭。调用前多数错误还能返回给旧程序；设置 `point_of_no_return` 后，旧映像可能已经被拆掉，后续失败只能让进程收到致命信号。

`de_thread()` 把线程组收敛为当前线程，符合官方文档“execve 后其他线程被销毁”。`exec_mmap()` 安装新 `mm` 并释放旧映射。`do_close_on_exec()` 关闭带 `FD_CLOEXEC` 的 fd。`flush_signal_handlers()` 重置捕获的信号处理。`commit_creds()` 让 setuid/setgid/capability 等新凭据生效。

第一性原理概述：exec 的难点是原子性：用户要么仍在旧程序里得到错误，要么永远进入新程序。`point_of_no_return` 是内核从“可回滚准备”切到“必须完成或杀死进程”的界线。

脚本：`#!` 如何变成解释器 exec

/* fs/binfmt_script.c */
static int load_script(struct linux_binprm *bprm)
{
        const char *i_name, *i_sep, *i_arg, *i_end, *buf_end;
        struct file *file;
        int retval;

        if ((bprm->buf[0] != '#') || (bprm->buf[1] != '!'))
                return -ENOEXEC;
        ...
        i_name = next_non_spacetab(bprm->buf+2, i_end);
        if (!i_name || (i_name == i_end))
                return -ENOEXEC;

        i_arg = NULL;
        i_sep = next_terminator(i_name, i_end);
        if (i_sep && (*i_sep != '\0'))
                i_arg = next_non_spacetab(i_sep, i_end);
        ...
        retval = remove_arg_zero(bprm);
        retval = copy_string_kernel(bprm->interp, bprm);
        bprm->argc++;
        if (i_arg) {
                retval = copy_string_kernel(i_arg, bprm);
                bprm->argc++;
        }
        retval = copy_string_kernel(i_name, bprm);
        bprm->argc++;
        retval = bprm_change_interp(i_name, bprm);
        ...
        file = open_exec(i_name);
        if (IS_ERR(file))
                return PTR_ERR(file);

        bprm->interpreter = file;
        return 0;
}

逐段解释：脚本加载器只识别 `#!`。它从第一行解析解释器路径和可选参数，检查脚本路径在 exec 后是否仍可访问，然后重写参数栈。

重写后的形态近似为：`interpreter [optional-arg] script-path old-argv[1...]`。随后 `bprm->interpreter` 指向解释器文件，`exec_binprm()` 循环会把解释器作为新的 `bprm->file` 再走一次 binfmt；通常解释器本身是 ELF。

第一性原理概述：脚本不是 CPU 能直接执行的指令映像。内核通过参数重写，把“执行脚本”转换成“执行解释器，并把脚本路径交给解释器”。

后半段枚举所有 `PT_LOAD` 段，把代码、只读数据、可写数据等映射进新 mm。`ET_EXEC` 使用固定地址；`ET_DYN` 会计算 `load_bias`，PIE 程序还会参与 ASLR。`p_filesz < p_memsz` 的尾部由 `set_brk()` 和清零逻辑建立 `.bss`/brk 区域。

若有解释器，最终入口不是主程序 `e_entry`，而是动态链接器的入口；主程序入口会通过 auxv 的 `AT_ENTRY` 等信息传给解释器。若无解释器，`elf_entry = e_entry`。

第一性原理概述：ELF loader 的工作是把文件中的“段描述”变成页表中的“虚拟内存区域”，再决定 CPU 从哪个虚拟地址开始取第一条用户态指令。

`setup_arg_pages()` 把临时参数 VMA 调整成真正用户栈，应用是否可执行、栈随机化和 RLIMIT_STACK 规则。`create_elf_tables()` 在栈上补齐 `argc`、`argv[]`、`envp[]`、auxv、平台字符串和 16 字节随机数。

auxv 是内核给动态链接器和 C 运行时的启动说明书，例如 `AT_PHDR`、`AT_PHENT`、`AT_ENTRY`、`AT_RANDOM`、`AT_PAGESZ` 等。动态链接器据此找到主程序 program header、入口、随机种子和硬件能力。

第一性原理概述：CPU 跳到入口地址时只知道寄存器和栈指针。新程序能拿到 `argc/argv/envp` 与动态链接信息，全靠内核在用户栈上预先摆好 ABI 约定的数据结构。

/* arch/mips/kernel/process.c */
void start_thread(struct pt_regs * regs, unsigned long pc, unsigned long sp)
{
        unsigned long status;

        status = regs->cp0_status & ~(ST0_CU0|ST0_CU1|ST0_CU2|ST0_FR|KU_MASK);
        status |= KU_USER;
        regs->cp0_status = status;
        lose_fpu(0);
        clear_thread_flag(TIF_MSA_CTX_LIVE);
        clear_used_math();
        ...
        regs->cp0_epc = pc;
        regs->regs[29] = sp;
}

`ELF_PLAT_INIT()` 把多数通用寄存器清零，避免旧程序寄存器状态泄漏给新程序。`finalize_exec()` 保存新的栈 rlimit。

`start_thread()` 切到用户权限 `KU_USER`，清理 FPU/MSA/DSP 等旧执行上下文，把 `cp0_epc` 设置成 `elf_entry`，把 `$sp` 即 `regs[29]` 设置成 `bprm->p`。之后系统调用返回路径恢复这组寄存器，CPU 从新程序入口开始在用户态运行。

第一性原理概述：exec 最终不是“调用”新程序，而是改写陷入内核时保存的用户寄存器现场。返回用户态那一刻，旧程序已经没有可返回的位置。

哪些保留，哪些重置

类别exec 后状态当前源码对应点
PID / task 身份保留，不创建新 PID`current` 一直是同一个 task；`trace_sched_process_exec(current, old_pid, bprm)` 记录 exec 事件。
地址空间旧用户映射丢弃，新 mm 安装`exec_mmap(bprm->mm)`，随后 ELF `elf_map()` 映射 `PT_LOAD`。
其他线程销毁，只剩 exec 线程`de_thread(me)`。
信号处理捕获处理器重置，默认/忽略按规则保留`flush_signal_handlers(me, 0)`。
文件描述符默认保留；`FD_CLOEXEC` 关闭`unshare_files()` 与 `do_close_on_exec(me->files)`。
凭据可能因 setuid/setgid/capability 改变`bprm_fill_uid()`、`security_bprm_creds_from_file()`、`commit_creds()`。
入口点改为 ELF 入口或解释器入口`start_thread(regs, elf_entry, bprm->p)`。
第一性原理概述：exec 保留“进程身份”以维持父子关系、PID、部分资源引用；重置“程序执行状态”以保证新程序从干净、可预测的 ABI 初始状态启动。

整体总结

入口

`execv()` 属于 libc exec 家族，内核主入口是 `execve` / `execveat`，都汇入 `do_execveat_common()`。

准备

内核用 `linux_binprm` 记录目标文件、解释器、凭据、argc/envc、临时参数栈和新 mm。

提交

`begin_new_exec()` 后进入不可回退点：去线程化、安装新 mm、关闭 CLOEXEC fd、提交新凭据、重置信号处理。

运行

ELF loader 映射 `PT_LOAD`、建立用户栈和 auxv；MIPS `start_thread()` 设置 EPC/SP，返回用户态即进入新程序。

一句话：fork 复制一个可调度身份，exec 把这个身份里的用户程序换掉。成功的 exec 没有返回值，因为旧程序已经不复存在。
