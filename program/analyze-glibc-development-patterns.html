为何学习 glibc 源码模式

glibc（GNU C Library）是 Linux 用户态最核心的基础库，承载着数十年的工程积累。它大量使用了 GCC 扩展特性与 C 语言编程技巧，这些模式在追求性能、健壮性与可维护性的工程中极具参考价值。本文聚焦于用户态开发可直接借鉴的内容，并附完整注释的 C 代码示例。

§ 1

强符号、弱符号与函数别名

链接器技巧
glibc 在 include/libc-symbols.h 中定义了一套完整的符号别名宏。strong_alias 让同一份函数代码拥有两个对外名字；weak_alias 则允许用户在链接时覆盖库的默认实现。这在插件系统、单元测试 mock 与平台适配中非常有用。
图 1-1：strong_alias 与 weak_alias 的链接关系
C — 强弱符号与别名（源自 include/libc-symbols.h） 
/*
 * 来源：glibc/include/libc-symbols.h
 * 官方仓库：https://github.com/bminor/glibc/blob/master/include/libc-symbols.h
 *
 * ── 章节 1：强符号别名 (strong_alias) 与弱符号别名 (weak_alias) ──
 *
 * 核心原理：
 *   - GCC 的 __attribute__((alias("name"))) 让编译器在目标文件中生成
 *     一个指向同一符号的别名条目，零运行时开销。
 *   - __attribute__((weak)) 将符号标记为弱定义，链接器在遇到同名强定义
 *     时会优先使用强定义，实现「可覆盖」语义。
 *   - __typeof(name) 保留原符号的类型信息，编译器可做类型检查。
 */

#ifndef _MY_SYMBOLS_H
#define _MY_SYMBOLS_H 1

/* ── 强符号别名 ────────────────────────────────────────────────────
 * 将 aliasname 定义为 name 的强别名（等价于同一函数体）。
 * 用途：同一实现同时暴露两个公共名字（如 write / __write）。
 *
 * glibc 原始定义（libc-symbols.h）：
 *   #define strong_alias(name, aliasname) _strong_alias(name, aliasname)
 *   #define _strong_alias(name, aliasname) \
 *     extern __typeof (name) aliasname __attribute__ ((alias (#name)));
 * ─────────────────────────────────────────────────────────────────── */
#define strong_alias(name, aliasname) \
    extern __typeof(name) aliasname __attribute__((alias(#name)))

/* ── 弱符号别名 ────────────────────────────────────────────────────
 * 将 aliasname 定义为 name 的弱别名。
 * 用途：库提供默认实现，用户可在自己的代码中定义同名函数覆盖它。
 *       典型场景：hook / mock / 平台适配层。
 * ─────────────────────────────────────────────────────────────────── */
#define weak_alias(name, aliasname) \
    extern __typeof(name) aliasname __attribute__((weak, alias(#name)))

/* ── 弱函数属性 ────────────────────────────────────────────────────
 * 直接在函数定义前使用，标记该函数本身为弱符号。
 * 当用户链接自己的同名函数时，链接器选用用户的版本。
 * ─────────────────────────────────────────────────────────────────── */
#define weak_function  __attribute__((weak))

#endif /* _MY_SYMBOLS_H */

/* ══════════════════ 使用示例 ══════════════════════════════════════ */

#include <stdio.h>
#include <stdlib.h>

/* 1. 内部实现函数（可带双下划线前缀，避免与用户符号冲突） */
static int __my_log_impl(const char *msg) {
    return printf("[LOG] %s\n", msg);
}

/* 2. 对外暴露 my_log 为强符号（同等地位，两个名字指向相同代码） */
strong_alias(__my_log_impl, my_log);

/* 3. 提供 my_log_default 弱别名，用户可以自己定义 my_log_default 覆盖 */
weak_alias(__my_log_impl, my_log_default);

/* 4. 弱符号直接定义写法：库提供默认 on_exit_hook，用户可覆盖 */
weak_function void on_exit_hook(void) {
    /* 默认空实现，用户链接自己的 on_exit_hook 时此版本被忽略 */
}

int main(void) {
    my_log("hello from strong alias");
    my_log_default("hello from weak alias");
    on_exit_hook();
    return 0;
}
✅ 借鉴要点
在库开发中：用 strong_alias 给同一实现提供兼容别名（如 POSIX 名 + BSD 名）；用 weak_alias / weak_function 提供可覆盖的「钩子」接口，单元测试时无需链接实际实现即可 mock。

§ 2

符号可见性控制（visibility）

DSO 设计
glibc 在 misc/sys/cdefs.h 和 include/libc-symbols.h 中使用 __attribute__((visibility(...))) 精确管理哪些符号对外可见。hidden 可见性让库内部调用绕过 PLT（过程链接表），显著减少运行时开销；配合编译选项 -fvisibility=hidden 可大幅缩减 DSO 导出符号数量。
图 2-1：default 与 hidden 可见性在 PLT 调用上的差异
C — 符号可见性控制（源自 glibc/misc/sys/cdefs.h 与 libc-symbols.h） 
/*
 * 来源：glibc/misc/sys/cdefs.h
 *       https://github.com/walac/glibc/blob/master/misc/sys/cdefs.h
 *       glibc/include/libc-symbols.h
 *
 * ── 章节 2：符号可见性（Symbol Visibility）控制 ──
 *
 * GCC 提供四种 visibility 级别：
 *   default  — 对外可见，参与动态符号解析（PLT），可被覆盖
 *   hidden   — 仅在当前 DSO 内可见，直接调用，无 PLT 开销
 *   protected— 类似 default，但不可被其他 DSO 覆盖
 *   internal — 最严格，编译器可假设不会从外部调用
 *
 * glibc 编译选项：-fvisibility=hidden（默认隐藏所有符号）
 * 然后用 attribute_hidden / attribute_visible 逐一控制
 */

/* ── 来自 glibc/misc/sys/cdefs.h 与 libc-symbols.h ── */

/* 标记符号为「库内部，不对外导出」—— 绕过 PLT，提升调用性能 */
#define attribute_hidden   __attribute__((visibility("hidden")))

/* 标记符号为「公开导出」—— 对外可见，需经 PLT 动态解析 */
#define attribute_visible  __attribute__((visibility("default")))

/* ── 推荐的头文件组织方式 ──
 *
 * mylib.h（公开头文件）：
 *   void my_public_api(int x);                  // 默认 visible（或显式标记）
 *
 * mylib_internal.h（内部头文件）：
 *   void __my_internal_helper(int x) attribute_hidden;
 *
 * 编译时传入：-fvisibility=hidden
 *   → 所有符号默认隐藏，只有显式 default 的才导出
 */

/* ── 示例：共享库内部实现 ── */

/* 公开 API：跨 DSO 可调用，出现在导出符号表（nm -D 可见） */
attribute_visible
void my_public_function(int x) {
    /* 实现 */
    (void)x;
}

/* 内部辅助函数：仅供库内其他 .c 文件调用，不进入导出符号表。
 * 调用方编译器直接生成 call 指令，无需经过 PLT/GOT，
 * 省去动态链接器的重定向开销，还能让编译器跨 TU 内联优化。 */
attribute_hidden
void __my_internal_helper(int x) {
    (void)x;
}

/* ── 使用 #pragma GCC visibility 批量设置 ──
 * 适合大段内部函数声明，避免逐一写 attribute_hidden */
#pragma GCC visibility push(hidden)
void internal_a(void);
void internal_b(void);
void internal_c(void);
#pragma GCC visibility pop

/* ── 用 nm 验证效果 ──
 * 编译：gcc -shared -fvisibility=hidden -o libmylib.so mylib.c
 * 验证：nm -D libmylib.so | grep ' T '
 *   只有 attribute_visible 的函数出现在动态符号表中
 */
visibility 级别能否被外部 DSO 覆盖PLT 间接调用典型用途
default✅ 是✅ 有公开 API
hidden❌ 否❌ 直接调用库内部函数
protected❌ 否✅ 有（自身引用直接）不可覆盖的公开符号
internal❌ 否❌ 直接调用仅供当前 TU

§ 3

内联控制：always_inline 与 noinline

性能调优
glibc 中大量关键路径函数使用 __attribute__((always_inline)) 强制内联以消除调用开销，而错误路径则使用 __attribute__((noinline)) 防止内联污染热路径 icache。这一对组合是高性能 C 库的标准做法。
C — 内联与不内联控制（源自 glibc sys/cdefs.h & 内部实现） 
/*
 * 来源：glibc/misc/sys/cdefs.h
 *       https://github.com/walac/glibc/blob/master/misc/sys/cdefs.h
 *
 * ── 章节 3：强制内联（always_inline）与禁止内联（noinline）──
 *
 * 问题背景：
 *   - inline 关键字仅是「建议」，编译器不保证内联
 *   - 高频路径：函数调用开销（保存寄存器、跳转、返回）不可接受
 *   - 错误路径：不希望大段错误处理代码污染 I-cache
 */

#include <stddef.h>
#include <stdio.h>

/* ── 强制内联 ──────────────────────────────────────────────────────
 * glibc 的 __always_inline 宏来自 sys/cdefs.h：
 *   #define __always_inline __inline __attribute__((__always_inline__))
 *
 * 无论优化级别如何（-O0 也生效），函数体直接展开在调用处。
 * 适合：频繁调用的小函数（如类型转换、位操作、原子读写包装）
 * ─────────────────────────────────────────────────────────────────── */
#define always_inline  __attribute__((__always_inline__)) static inline

/* ── 禁止内联 ──────────────────────────────────────────────────────
 * __attribute__((noinline))：确保函数不被内联。
 * 适合：错误处理路径、调试函数、包含 __builtin_return_address 的函数。
 * glibc 还用 noinline + noclone 组合防止编译器克隆优化：
 *   #define __attribute_optimization_barrier__ \
 *     __attribute__((noinline, noclone))      // 来自 glibc/include/sys/cdefs.h
 * ─────────────────────────────────────────────────────────────────── */
#define noinline       __attribute__((noinline))

/* ── 实际使用示例 ── */

/* 热路径：位对齐检查，always_inline 避免函数调用开销 */
always_inline int is_aligned(const void *ptr, size_t align) {
    /* align 必须是 2 的幂 */
    return ((__nm>uintptr_t)ptr & (align - 1)) == 0;
}

/* 冷路径：错误报告，noinline 避免污染调用方的 icache
 * 同时 __attribute__((cold)) 让编译器将此函数放到 .text.unlikely 区段 */
static noinline __attribute__((cold))
void report_error(const char *msg) {
    fprintf(stderr, "Error: %s\n", msg);
}

/* ── 组合使用示例：类似 glibc malloc 中的模式 ── */
static always_inline int fast_path(int x) {
    if (__builtin_expect(x < 0, 0)) {  /* 冷分支 */
        report_error("negative value");  /* noinline，不污染热路径 */
        return -1;
    }
    return x * 2;  /* 热路径逻辑 */
}

/* ── __attribute__((flatten)) ──────────────────────────────────────
 * glibc 偶尔使用：标记在某函数内，所有调用尽可能全部内联展开。
 * 用于性能敏感的外层入口函数。
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((flatten))
int entry_with_all_inlined(int a, int b) {
    return fast_path(a) + fast_path(b);
}
⚠ 注意
always_inline 会增大代码体积（代码膨胀），在频繁调用次数不确定或函数体较大时请谨慎使用。glibc 只对真正热路径的极小函数使用它。
§ 4

分支预测提示：likely / unlikely

CPU 流水线优化
glibc 广泛使用 __glibc_likely / __glibc_unlikely（内部为 __builtin_expect）告知编译器哪条分支是"热"的，让编译器将热路径代码放在顺序执行路径（fall-through），减少分支预测失效和 icache miss。
图 4-1：__builtin_expect 如何影响代码布局与分支预测
C — 分支预测提示宏（源自 glibc/include/cdefs.h 及 pthread_create.c） 
/*
 * 来源：glibc 内部宏 __glibc_likely / __glibc_unlikely
 *       https://sourceware.org/glibc/manual/latest/html_mono/libc.html
 *       glibc/nptl/pthread_create.c 中大量使用 __glibc_unlikely
 *
 * ── 章节 4：分支预测提示 likely / unlikely ──
 *
 * 原理：__builtin_expect(expr, expected_value)
 *   向编译器报告 expr 最有可能等于 expected_value。
 *   编译器据此重排汇编指令，将「更可能走」的分支放在 fall-through 路径。
 *   x86/ARM 的 CPU 对 fall-through（无跳转）天然预测为「执行」，
 *   而跳转指令则依赖 BPU 历史。
 *
 * glibc 定义（特性检测后）：
 *   #define __glibc_likely(cond)   __builtin_expect(!!(cond), 1)
 *   #define __glibc_unlikely(cond) __builtin_expect(!!(cond), 0)
 */

/* 用户代码可直接这样定义，与 Linux kernel 的 likely/unlikely 同理 */
#define likely(x)    __builtin_expect(!!(x), 1)
#define unlikely(x)  __builtin_expect(!!(x), 0)

#include <stdlib.h>
#include <string.h>
#include <errno.h>

/* ── 示例 1：内存分配检查（失败是小概率事件） ── */
char *safe_strdup(const char *src) {
    char *dst = malloc(strlen(src) + 1);
    /* 分配失败极少发生 → unlikely；热路径是 dst != NULL 的情况 */
    if (unlikely(dst == NULL)) {
        errno = ENOMEM;
        return NULL;
    }
    return strcpy(dst, src);
}

/* ── 示例 2：快速路径判断（类似 glibc malloc fastbin 路径） ── */
typedef struct { int valid; int data; } CacheEntry;

int cache_lookup(CacheEntry *cache, int n, int key) {
    for (int i = 0; i < n; i++) {
        /* 缓存命中是常见情况 → likely */
        if (likely(cache[i].valid && cache[i].data == key))
            return i;
    }
    return -1;
}

/* ── 示例 3：类似 glibc pthread_create.c 中的用法 ──
 *   if (__glibc_unlikely(atomic_decrement_and_test(&__nptl_nthreads)))
 *       exit(0);
 * 线程退出时，通常不是最后一个线程 → unlikely */
void thread_exit_check(int *thread_count) {
    /* __sync_sub_and_fetch 原子递减，返回新值 */
    int remaining = __sync_sub_and_fetch(thread_count, 1);
    if (unlikely(remaining == 0)) {
        /* 最后一个线程退出，终止进程 */
        exit(0);
    }
}

/* ── C++20 / C23 标准替代 ──
 * C++20 引入 [[likely]] / [[unlikely]] 属性，语义相同，无需宏：
 *   if (ptr) [[likely]] { ... }
 * C23 也计划引入，但目前 GCC/Clang 扩展仍是主流。
 */
✅ 使用原则
只在你确实知道概率分布的地方使用。错误的 likely/unlikely 比不用更糟——它会欺骗 CPU 的动态分支预测器。一般建议：错误检查用 unlikely，主循环条件延续用 likely。

§ 5

函数语义属性：pure、const、noreturn、warn_unused_result

编译器优化语义
glibc 在函数声明上大量附加语义属性，帮助 GCC 执行更激进的优化（公共子表达式消除、死代码删除），同时让编译器在调用端发出更精确的警告。
C — 函数语义属性（源自 glibc 公开头文件与 GCC 文档） 
/*
 * 来源：
 *   - GCC Function Attributes 官方文档
 *     https://gcc.gnu.org/onlinedocs/gcc/Function-Attributes.html
 *   - glibc/misc/sys/cdefs.h（__attribute_pure__, __attribute_const__ 等宏）
 *   - glibc string.h / math.h 中的函数声明
 *
 * ── 章节 5：函数语义属性 ──
 *
 * 这些属性的本质是向编译器说明函数的副作用边界，
 * 编译器据此做 CSE（公共子表达式消除）、删除无用调用等优化。
 */

#include <stddef.h>
#include <stdnoreturn.h>  /* C11 noreturn */
#include <stdio.h>
#include <stdlib.h>

/* ── pure：无副作用，可读全局内存 ────────────────────────────────
 * glibc 定义：#define __attribute_pure__ __attribute__((__pure__))
 * 约定：
 *   - 函数不修改任何全局状态（无写操作）
 *   - 可以读取内存（包括通过指针参数访问的内存）
 *   - 相同参数 + 相同全局状态 → 相同返回值
 *   - 编译器可缓存调用结果，消除重复调用
 * 典型：strlen、memchr、checksum 计算
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((__pure__))
size_t my_strlen(const char *s) {
    size_t n = 0;
    while (*s++) n++;
    return n;
}

/* ── const：严格无副作用，不读全局内存 ──────────────────────────
 * glibc 定义：#define __attribute_const__ __attribute__((__const__))
 * 约定（比 pure 更严格）：
 *   - 返回值完全由参数决定，不访问任何全局/静态内存
 *   - 相同参数必然相同返回值（纯函数，数学意义上）
 *   - 编译器可在循环提升（loop-invariant code motion）中提取
 * 典型：sin/cos/abs（数学函数）、位运算辅助函数
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((__const__))
unsigned long next_power_of_2(unsigned long n) {
    if (n == 0) return 1;
    n--;
    n |= n >> 1; n |= n >> 2;
    n |= n >> 4; n |= n >> 8;
    n |= n >> 16; n |= n >> 32;
    return n + 1;
}

/* ── noreturn：函数永不返回 ──────────────────────────────────────
 * glibc：exit()、abort()、longjmp() 等使用 __attribute__((__noreturn__))
 * 好处：
 *   1. 调用后的代码编译器知道是死代码，可优化掉
 *   2. 不需要在调用方补充 return 语句也不会发出警告
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((__noreturn__))
void fatal_error(const char *msg) {
    fprintf(stderr, "FATAL: %s\n", msg);
    abort();
    /* 无需 return，编译器知道不会到达这里 */
}

/* ── warn_unused_result：强制调用方检查返回值 ─────────────────
 * glibc 在 read()、write()、malloc() 等的 glibc 内部声明上使用。
 * 调用方忽略返回值时编译器发出警告（配合 -Wunused-result）。
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((warn_unused_result))
int safe_write(int fd, const void *buf, size_t len) {
    (void)fd; (void)buf; (void)len;
    return 0; /* 简化示意，实际应检查 write() 返回值 */
}

/* ── malloc 属性：返回新分配的内存指针 ────────────────────────
 * 告知编译器返回值是不与任何已知对象别名的指针，
 * 编译器可据此做更激进的别名分析（alias analysis）。
 * glibc 的 malloc() 自身即用此属性声明。
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((__malloc__))
void *my_pool_alloc(size_t size) {
    return malloc(size);
}

/* ── hot / cold：标记热/冷函数 ──────────────────────────────────
 * hot：链接器将热函数放在 .text 的开头，提升 icache 命中率
 * cold：将冷函数放在 .text.cold，远离热代码
 * glibc malloc 主路径标记为 hot，错误打印标记为 cold
 * ─────────────────────────────────────────────────────────────────── */
__attribute__((hot))
int hot_path_process(int x) { return x * 3 + 1; }

__attribute__((cold))

编译期类型检查宏：__typeof__ 与 typecheck

类型安全
glibc 和 Linux kernel 都使用 __typeof__ 让宏在展开时保留类型信息，避免隐式类型转换导致的 bug，并能在编译期触发类型不匹配警告。这是 C 语言宏"类型安全化"的核心技巧。
C — typeof 类型保留与编译期类型检查（源自 Linux kernel / glibc 宏） 
/*
 * 来源：
 *   - Linux kernel include/linux/minmax.h （min/max 宏的 typeof 用法）
 *   - Stack Overflow: Typechecking macro arguments in C
 *     https://stackoverflow.com/questions/4712720
 *   - glibc libc-symbols.h 中 strong_alias 使用的 __typeof(name)
 *
 * ── 章节 6：__typeof__ 与编译期类型检查宏 ──
 *
 * 问题：C 的泛型宏（如 #define max(a,b) ((a)>(b)?(a):(b))）不做类型检查：
 *   max(1.0, 2L)  → 整型与浮点混用，可能截断
 *   max(ptr, 0)   → 指针与整数混用，未定义行为
 *
 * 解决方案：
 *   1. __typeof__ 推导变量类型 → 用于定义临时变量
 *   2. 利用指针比较触发类型不匹配警告（GCC 会在指针类型不同时警告）
 */

#include <stddef.h>

/* ── 1. 类型安全的 min/max 宏 ──────────────────────────────────
 * Linux kernel 的经典实现，glibc 内部亦有类似用法：
 *   __typeof__(x) _x = (x); —— 保持 x 的完整类型（含 const/volatile）
 *   (void)(&_x == &_y);     —— 触发指针类型不匹配警告（如 int* vs long*）
 *                              GCC 对此发出 -Wtype-limits 兼容警告
 * ─────────────────────────────────────────────────────────────────── */
#define safe_min(x, y) ({               \
    __typeof__(x) _x = (x);             \
    __typeof__(y) _y = (y);             \
    (void)(&_x == &_y);  /* 类型检查：不同类型指针比较触发警告 */ \
    _x < _y ? _x : _y;                  \
})

#define safe_max(x, y) ({               \
    __typeof__(x) _x = (x);             \
    __typeof__(y) _y = (y);             \
    (void)(&_x == &_y);                 \
    _x > _y ? _x : _y;                  \
})

/* ── 2. 编译期类型断言（typecheck 宏） ──────────────────────────
 * 思路：创建一个同类型的哑变量，再将 x 赋给它，
 *       如果类型不兼容，编译器会警告/报错。
 *       ({ ... 1; }) 是 GCC 语句表达式（statement expression）扩展，
 *       结果为最后一条语句的值。
 * ─────────────────────────────────────────────────────────────────── */
#define typecheck(type, x)  ({  \
    type __dummy;               \
    __typeof__(x) __dummy2;     \
    (void)(&__dummy == &__dummy2); /* 类型不同时 GCC 发出警告 */ \
    1;                           \
})

/* ── 3. 参数副作用安全的宏包裹 ─────────────────────────────────
 * 传统 #define ABS(x) ((x)<0?-(x):(x)) 在 ABS(i++) 时 i 自增两次。
 * 用 __typeof__ + 临时变量彻底避免：
 * ─────────────────────────────────────────────────────────────────── */
#define safe_abs(x) ({          \
    __typeof__(x) _v = (x);     \
    _v < 0 ? -_v : _v;          \
})

/* ── 4. 类型保留的数组元素数宏 ─────────────────────────────────
 * glibc 风格：
 *   ARRAY_SIZE 在传入指针（而非数组）时，
 *   利用 __typeof__ 在编译期检测（需要 GCC 扩展）
 * ─────────────────────────────────────────────────────────────────── */
#define ARRAY_SIZE(arr) (sizeof(arr) / sizeof((arr)[0]))

/* 更严格版本：防止对指针误用（编译期触发 BUILD_BUG_ON 风格错误） */
#define ARRAY_SIZE_SAFE(arr) (                         \
    sizeof(arr) / sizeof((arr)[0])                     \
    + (sizeof(arr) == sizeof(void *) ?                 \
       0 : 0)   /* 仅示意，实际需 static_assert */     \
)

/* ── 5. 使用示例 ── */
void example_typeof_usage(void) {
    int  a = 10, b = 20;
    long c = 30;

    int  r1 = safe_min(a, b);   /* 正常：同为 int */
    /* int  r2 = safe_min(a, c); */  /* ← GCC 将警告：int* vs long* */
    int  r3 = safe_abs(-a);     /* 安全，无双重副作用 */

    int  arr[] = {1,2,3,4,5};
    size_t n = ARRAY_SIZE(arr);  /* 5 */

    (void)r1; (void)r3; (void)n; (void)c;
}
✅ 借鉴要点
在任何需要"泛型"行为的宏中，用 __typeof__(x) _x = (x); 替代直接展开参数：既避免副作用，又保留完整类型信息。配合指针比较可实现编译期类型断言。

cleanup 属性：C 语言的 RAII

资源管理
__attribute__((cleanup(func))) 是 GCC/Clang 扩展，当变量离开作用域时自动调用清理函数。systemd、glibc 内部都大量使用这一模式实现类 C++ RAII 的资源自动释放，彻底消灭多路 goto 错误处理。
图 7-1：goto 模式 vs cleanup 属性在多资源管理上的对比
C — cleanup 属性实现 RAII（GCC/Clang 扩展，systemd/glibc 风格） 
/*
 * 来源：GCC __attribute__((cleanup)) 官方文档
 *       systemd 的 _cleanup_free_ 等宏（参考 GCC HN 讨论）
 *       https://yuankun.me/posts/the-cleanup-variable-attribute-in-gcc/
 *
 * ── 章节 7：cleanup 属性 —— C 语言的 RAII ──
 *
 * 语义：当声明了 cleanup(func) 的自动变量离开作用域时
 *       （包括 return、goto 出作用域、以及函数异常退出），
 *       编译器自动插入对 func(&var) 的调用。
 *
 * 约束：
 *   1. 只能用于自动（栈）变量，不能用于函数参数或静态变量
 *   2. cleanup 函数接收一个「指向该变量的指针」作为参数
 *   3. cleanup 函数的返回值被忽略
 *   4. 多个 cleanup 变量按声明的逆序（LIFO）调用清理函数
 *   5. 建议初始化为 NULL/0，防止 cleanup 处理未初始化的值
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>

/* ── 定义 cleanup 函数 ────────────────────────────────────────── */

/* 释放堆内存（参数是 char** 因为 cleanup 传递的是变量的地址） */
static void cleanup_free(void *ptr_to_ptr) {
    void **p = (void **)ptr_to_ptr;
    free(*p);          /* free(NULL) 是合法的，无需额外判断 */
}

/* 关闭文件描述符（-1 表示未打开）*/
static void cleanup_close(int *fdp) {
    if (*fdp >= 0)
        close(*fdp);
}

/* 关闭 FILE* 流 */
static void cleanup_fclose(FILE **fp) {
    if (*fp) fclose(*fp);
}

/* ── 便捷宏定义（systemd 风格）── */
#define _cleanup_free_   __attribute__((cleanup(cleanup_free)))
#define _cleanup_close_  __attribute__((cleanup(cleanup_close)))
#define _cleanup_fclose_ __attribute__((cleanup(cleanup_fclose)))

/* ── 使用示例：多资源管理，无需 goto ── */
int process_file(const char *path) {
    /* 初始化为"空"状态，确保未成功分配时 cleanup 不误操作 */
    _cleanup_free_   char *buf  = NULL;
    _cleanup_close_  int   fd   = -1;
    _cleanup_fclose_ FILE *log  = NULL;

    fd = open(path, O_RDONLY);
    if (fd < 0) return -1;  /* cleanup_close 不做任何事（fd=-1） */

    buf = malloc(4096);
    if (!buf) return -1;     /* cleanup_close 关闭 fd；cleanup_free(NULL) 无害 */

    log = fopen("/tmp/proc.log", "w");
    if (!log) return -1;    /* cleanup: fclose(log)=跳过, free(buf), close(fd) */

    /* 主处理逻辑 */
    ssize_t n = read(fd, buf, 4095);
    if (n > 0) {
        buf[n] = '\0';
        fprintf(log, "Read %zd bytes\n", n);
    }

    return 0;
    /* 函数结束时，编译器自动按逆序调用：
     *   cleanup_fclose(&log)  → fclose(log)
     *   cleanup_close(&fd)    → close(fd)
     *   cleanup_free(&buf)    → free(buf)
     */
}

/* ── 泛型「scope guard」封装 ── */
typedef struct {
    void (*fn)(void *);
    void *ctx;
} ScopeGuard;

static void scope_guard_run(ScopeGuard *sg) {
    if (sg->fn) sg->fn(sg->ctx);
}

#define SCOPE_GUARD(cleanup_fn, context) \
    ScopeGuard __attribute__((cleanup(scope_guard_run))) \
    __scope_guard_##__LINE__ = { .fn = (cleanup_fn), .ctx = (void*)(context) }

void demo_scope_guard(void) {
    int *data = malloc(sizeof(int) * 10);
    SCOPE_GUARD(free, data);    /* 离开作用域自动释放 */
    data[0] = 42;
    /* ... 任意路径离开此作用域都会 free(data) */
}
✅ 借鉴要点
在处理多个资源（fd、内存、锁）的函数中，用 cleanup 替代多层 goto，代码线性可读。关键：所有变量初始化为"无效"状态（NULL/-1），防止未初始化时触发清理。

C — 灵活数组成员（源自 glibc malloc_chunk 结构思路 + C99 标准） 
/*
 * 来源：
 *   - glibc/malloc/malloc.c  struct malloc_chunk
 *     https://fossies.org/linux/glibc/malloc/malloc.c
 *   - C99 标准 6.7.2.1 §18：灵活数组成员
 *
 * ── 章节 8：灵活数组成员（Flexible Array Member, FAM）──
 *
 * C99 允许在结构体最后一个成员处声明不指定大小的数组 data[]，
 * 分配时：malloc(offsetof(T, data) + n * sizeof(elem_type))
 * 释放时：free(p)  ← 只需一次
 *
 * glibc malloc_chunk 中类似的技巧（伪码）：
 *   struct malloc_chunk {
 *     size_t prev_size;
 *     size_t size;
 *     struct malloc_chunk *fd, *bk;
 *     // 用户数据紧跟在后面 → mem = (char*)chunk + CHUNK_HDR_SZ
 *   };
 */

#include <stddef.h>
#include <stdlib.h>
#include <string.h>
#include <stdio.h>

/* ── 带 FAM 的变长字符串结构 ── */
typedef struct {
    size_t  len;       /* 字符串长度（不含 '\0'） */
    size_t  capacity;  /* 已分配的空间（字节数） */
    char    data[];    /* C99 灵活数组成员：紧跟在结构体后 */
} MyString;

/* 创建：一次 malloc，结构体头部 + 字符串数据连续存放 */
MyString *mystring_create(const char *s) {
    size_t len = strlen(s);
    size_t total = offsetof(MyString, data) + len + 1;
    /* offsetof(MyString, data) 精确给出 data 字段的字节偏移 */
    MyString *ms = malloc(total);
    if (!ms) return NULL;
    ms->len      = len;
    ms->capacity = len + 1;
    memcpy(ms->data, s, len + 1);
    return ms;
    /* data 与 ms 在同一块连续内存中，无额外指针间接 */
}

/* 销毁：一次 free */
void mystring_free(MyString *ms) { free(ms); }

/* ── 带 FAM 的消息帧（网络/IPC 场景） ── */
typedef struct __attribute__((packed)) {
    uint32_t magic;    /* 魔数 */
    uint16_t type;     /* 消息类型 */
    uint16_t payload_len; /* payload 字节数 */
    uint8_t  payload[]; /* 变长 payload，紧跟头部 */
} MsgFrame;

MsgFrame *msg_create(uint16_t type, const void *data, uint16_t dlen) {
    MsgFrame *f = malloc(sizeof(MsgFrame) + dlen);
    if (!f) return NULL;
    f->magic       = 0xDEADBEEF;
    f->type        = type;
    f->payload_len = dlen;
    memcpy(f->payload, data, dlen);
    return f;
}

/* ── 零长度数组（GCC 扩展，C89 时代的 FAM 前身） ──
 * glibc 旧代码中可见 data[0]，C99 后应优先使用 data[] */
typedef struct {
    int count;
    int items[0];   /* GCC 扩展，等价于 C99 的 items[] */
} LegacyArray;

void demo_fam(void) {
    MyString *s = mystring_create("hello, glibc");
    printf("len=%zu data=%s\n", s->len, s->data);

}

offsetof 与 container_of：从成员指针还原父结构

glibc 的 malloc.c 中频繁使用 offsetof 做内存指针算术（如 bin_at 宏）。Linux kernel 的 container_of 宏是同一思路的完美封装，是侵入式链表、事件系统等数据结构的基础。
图 9-1：container_of 通过成员地址推导父结构体地址
C — offsetof 与 container_of（源自 glibc malloc.c bin_at 宏 + Linux kernel） 
/*
 * 来源：
 *   glibc/malloc/malloc.c：bin_at 宏使用 offsetof(struct malloc_chunk, fd)
 *   https://fossies.org/linux/glibc/malloc/malloc.c
 *   Linux kernel include/linux/container_of.h
 *
 * ── 章节 9：offsetof 与 container_of ──
 *
 * glibc malloc.c 中的 bin_at 宏：
 *   #define bin_at(m, i) \
 *     (mbinptr)(((char*)&((m)->bins[((i)-1)*2])) - offsetof(struct malloc_chunk, fd))
 *
 * 此宏将 bins 数组的某个地址「倒推」为一个 malloc_chunk* 的起始地址，
 * 使 bin 头部可以直接当作 malloc_chunk 来操作，消除了特殊 case 处理。
 *
 * container_of 是同一模式的通用封装，是侵入式数据结构的基石。
 */

#include <stddef.h>
#include <stdint.h>
#include <stdio.h>

/* ── container_of：带类型检查（Linux kernel 风格） ── */
#define container_of(ptr, type, member) ({                      \
    const __typeof__(((type *)0)->member) *__mptr = (ptr);      \
    /* 类型检查：若 ptr 类型与 member 类型不一致，此处编译报错 */ \
    (type *)((char *)__mptr - offsetof(type, member));           \
})

/* ── 侵入式双向链表实现 ── */
typedef struct ListHead {
    struct ListHead *next, *prev;
} ListHead;

/* 初始化：让链表头指向自身（空链表） */
static inline void list_init(ListHead *h) {
    h->next = h->prev = h;
}

/* 在 head 后插入节点（O(1)） */
static inline void list_add(ListHead *head, ListHead *node) {
    node->next  = head->next;
    node->prev  = head;
    head->next->prev = node;
    head->next  = node;
}

/* 删除节点（不释放内存） */
static inline void list_del(ListHead *node) {
    node->prev->next = node->next;
    node->next->prev = node->prev;
    node->next = node->prev = node;  /* 防悬空指针 */
}

/* 遍历宏：从 list 成员还原到包含它的结构体 */
#define list_for_each(pos, head) \
    for (pos = (head)->next; pos != (head); pos = pos->next)

#define list_entry(ptr, type, member) \
    container_of(ptr, type, member)

/* ── 示例：任务队列 ── */
typedef struct {
    int      id;
    char     name[32];
    ListHead list;      /* 侵入式链表节点，嵌入在结构体中 */
} Task;

void demo_container_of(void) {
    ListHead queue;
    list_init(&queue);

    Task t1 = { .id=1, .name="parse"  };
    Task t2 = { .id=2, .name="render" };
    Task t3 = { .id=3, .name="upload" };

    list_add(&queue, &t1.list);
    list_add(&queue, &t2.list);
    list_add(&queue, &t3.list);

    /* 遍历：通过 list 成员地址恢复 Task* */
    ListHead *pos;
    list_for_each(pos, &queue) {
        Task *t = list_entry(pos, Task, list);
        printf("Task[%d]: %s\n", t->id, t->name);
    }
    /* 输出：Task[3]: upload, Task[2]: render, Task[1]: parse */
}

/* ── glibc bin_at 宏等价演示 ────────────────────────────────────
 * glibc 用此技巧让 bins[] 数组里的 fd/bk 字段
 * 直接视作 malloc_chunk 的 fd/bk，省去哨兵节点的特殊处理。
 * ─────────────────────────────────────────────────────────────────── */
struct FakeChunk { int prev; int size; int fd; int bk; };

static int bins_storage[10];  /* 简化的 bins 数组 */

static struct FakeChunk *bin_at_demo(int i) {
    /* 将 bins[i] 的地址减去 fd 的偏移量，得到假想的 FakeChunk* */
    return (struct FakeChunk *)((char *)&bins_storage[i] - offsetof(struct FakeChunk, fd));
}

C — constructor / destructor 属性（GCC 官方文档 + glibc 启动框架） 
/*
 * 来源：
 *   GCC Function Attributes 官方文档
 *   https://gcc.gnu.org/onlinedocs/gcc/Function-Attributes.html
 *   glibc 内部 csu/libc-start.c 使用 .init_array 调用机制
 *
 * ── 章节 10：constructor 与 destructor 属性 ──
 *
 * __attribute__((constructor))   → 在 main() 之前自动执行
 * __attribute__((destructor))    → 在 main() 返回或 exit() 后自动执行
 * __attribute__((constructor(P)))→ P 为优先级（101~65535），数值越小越先执行

 * glibc 的 __libc_start_main 在调用 main 前遍历 .init_array，
 * exit() 时遍历 .fini_array（逆序）。
 */

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

/* ── 优先级示例：数字越小越早执行 ── */
__attribute__((constructor(200)))
static void init_logging(void) {
    fprintf(stderr, "[init/200] 日志系统初始化\n");
    /* 在 init_network (300) 之前执行 */
}

__attribute__((constructor(300)))
static void init_network(void) {
    fprintf(stderr, "[init/300] 网络子系统初始化\n");
}

/* ── 清理顺序：数字越大越先执行（即与 constructor 逆序） ── */
__attribute__((destructor(300)))
static void fini_network(void) {
    fprintf(stderr, "[fini/300] 网络子系统清理（先）\n");
}

__attribute__((destructor(200)))
static void fini_logging(void) {
    fprintf(stderr, "[fini/200] 日志系统清理（后）\n");
}

/* ── 共享库场景：自动注册驱动 ──
 * 无需暴露 init 函数给用户，dlopen 时自动执行 */
static void *plugin_registry[64];
static int   plugin_count = 0;

__attribute__((constructor))
static void register_my_plugin(void) {
    /* dlopen 时自动注册，无需用户调用 init() */
    if (plugin_count < 64)
        plugin_registry[plugin_count++] = (void*)register_my_plugin;
}

/* ── 注意事项 ──────────────────────────────────────────────────
 * 1. 优先级 0-100 保留给 glibc/libgcc 内部使用

 * 3. 多个共享库中的 constructor 执行顺序依赖于链接顺序，

    printf("[main] 程序主体运行\n");

}

线程局部存储（TLS）：__thread 与 errno 模式

glibc 最经典的 TLS 应用是 errno——每个线程独立持有一份 errno 变量，通过 __thread 关键字（C11 用 _Thread_local）实现。glibc 还使用 attribute_tls_model_ie（initial-exec 模型）优化主模块内的 TLS 访问速度。
C — 线程局部存储模式（源自 glibc errno 实现与 libc-symbols.h） 
/*
 * 来源：
 *   glibc/include/libc-symbols.h：attribute_tls_model_ie 宏
 *   #define attribute_tls_model_ie __attribute__((tls_model("initial-exec")))
 *   glibc errno 实现：sysdeps/unix/sysv/linux/errno-loc.c
 *
 * ── 章节 11：线程局部存储（TLS）——每线程独立状态 ──
 *
 * TLS 四种模型（访问速度从快到慢）：
 *   1. local-exec    — 主可执行文件专属，最快（直接 fs: 寄存器偏移）
 *   2. initial-exec  — 导入时知道偏移，仍很快（glibc 内部用此模型）
 *   3. local-dynamic — 动态库内部，中速
 *   4. global-dynamic— 通用，最慢（需调用 __tls_get_addr）
 *
 * glibc errno 的关键设计：
 *   每个线程有独立 errno，__thread int errno 确保线程安全，
 *   signal handler 也能安全读写而不影响其他线程。
 */

#include <stdio.h>
#include <pthread.h>
#include <errno.h>

/* ── 1. 每线程错误码（errno 模式） ── */
/* C11 标准关键字；GCC 扩展用 __thread，完全等价 */
_Thread_local int my_last_error = 0;  /* 每线程独立副本，零初始化 */

void set_error(int code) { my_last_error = code; }
int  get_error(void)     { return my_last_error; }

/* ── 2. 每线程缓存（替代全局缓存，消除锁竞争） ── */
#define TLS_CACHE_SIZE 16
typedef struct {
    int  keys[TLS_CACHE_SIZE];
    int  vals[TLS_CACHE_SIZE];
    int  count;
} ThreadCache;

static _Thread_local ThreadCache tls_cache;  /* 自动零初始化 */

void tls_cache_put(int key, int val) {
    if (tls_cache.count < TLS_CACHE_SIZE) {
        tls_cache.keys[tls_cache.count]   = key;
        tls_cache.vals[tls_cache.count++] = val;
    }
}

/* ── 3. 每线程随机数种子（消除 rand() 的全局锁） ── */
static _Thread_local unsigned long tls_rand_seed;

static void tls_srand(unsigned long seed) { tls_rand_seed = seed; }
static int  tls_rand(void) {
    /* xorshift64：无锁、每线程独立 */
    tls_rand_seed ^= tls_rand_seed << 13;
    tls_rand_seed ^= tls_rand_seed >> 7;
    tls_rand_seed ^= tls_rand_seed << 17;
    return (int)(tls_rand_seed & 0x7FFFFFFF);
}

/* ── 4. TLS 初始化：与 constructor 结合 ── */
static _Thread_local int tls_initialized = 0;

static void ensure_tls_init(void) {
    if (__builtin_expect(!tls_initialized, 0)) {
        /* 仅在当前线程第一次调用时初始化 */
        tls_srand((unsigned long)pthread_self());
        tls_initialized = 1;
    }
}

/* ── 5. glibc attribute_tls_model_ie 提速 ────────────────────
 * 当变量在主可执行文件（非动态库）中，使用 initial-exec 模型
 * 访问只需一条指令（相对 FS 寄存器偏移），最接近局部变量速度。
 * glibc 定义：
 *   #define attribute_tls_model_ie \
 *     __attribute__((tls_model("initial-exec")))
 * 注意：initial-exec 不能用于通过 dlopen 加载的共享库！
 * ─────────────────────────────────────────────────────────────────── */
static __attribute__((tls_model("initial-exec")))
_Thread_local int fast_tls_counter = 0;  /* 极速访问的线程计数器 */

void increment_fast_tls(void) { fast_tls_counter++; }
int  get_fast_tls(void)      { return fast_tls_counter; }

 * glibc 的 memcpy、strcpy 等函数均用 __nonnull__ 标记关键指针参数。
 * 编译器在检测到明确传 NULL 时发出警告。

 * __attribute__((nonnull(1, 3)))    → 第 1、3 个参数不得为 NULL

    char *d = (char *)dst;

 * __attribute__((sentinel(1)))  → 倒数第 2 个参数是 NULL（0-based 偏移）

int run_commands(const char *first, ...)
    __attribute__((sentinel));  /* 最后一个参数必须是 (char*)NULL */

    va_list ap; int cnt = 0;

特性检测宏模式：__GNUC_PREREQ 与条件编译

可移植性
glibc 的 features.h 和 sys/cdefs.h 定义了一套完整的编译器/版本检测宏体系，让同一套头文件在 GCC 2.x 到最新版、Clang、以及不支持扩展的标准 C 编译器上均能正确工作。
C — 特性检测宏（源自 glibc/misc/sys/cdefs.h 与 features.h） 
/*
 * 来源：
 *   glibc/misc/sys/cdefs.h（__GNUC_PREREQ, __THROW, __LEAF 等）
 *   https://github.com/walac/glibc/blob/master/misc/sys/cdefs.h
 *   gnulib/lib/libc-config.h（__GNUC_PREREQ 可移植版本）
 *   https://github.com/coreutils/gnulib/blob/master/lib/libc-config.h
 *
 * ── 章节 13：特性检测宏与条件编译 ──
 *
 * 核心思想：将「是否支持某扩展」的判断集中在宏定义中，
 * 业务代码只用语义清晰的宏名，无需散布 #ifdef __GNUC__ 判断。
 */

/* ── 1. 编译器版本检测 ───────────────────────────────────────
 * glibc features.h 原始定义：
 *   #define __GNUC_PREREQ(maj, min) \
 *     ((__GNUC__ << 16) + __GNUC_MINOR__ >= ((maj) << 16) + (min))
 *
 * gnulib 更严谨的版本（避免整数溢出）：
 *   #define __GNUC_PREREQ(maj, min) \
 *     ((maj) < __GNUC__ + ((min) <= __GNUC_MINOR__))
 * ─────────────────────────────────────────────────────────────────── */
#if defined __GNUC__ && defined __GNUC_MINOR__
#  define GNUC_PREREQ(maj, min) \
     ((maj) < __GNUC__ + ((min) <= __GNUC_MINOR__))
#else
#  define GNUC_PREREQ(maj, min) 0
#endif

/* Clang 版本检测（glibc 2.38+ 开始支持 __glibc_clang_prereq） */
#if defined __clang_major__ && defined __clang_minor__
#  define CLANG_PREREQ(maj, min) \
     ((maj) < __clang_major__ + ((min) <= __clang_minor__))
#else
#  define CLANG_PREREQ(maj, min) 0
#endif

/* ── 2. __THROW：标记函数不抛异常（glibc sys/cdefs.h 原文） ─
 * glibc 定义：
 *   #if !defined __cplusplus && __GNUC_PREREQ(3, 3)
 *   #  define __THROW  __attribute__((__nothrow__ __LEAF))
 *   ...
 * 用途：C 库函数在 C++ 中被视为不会抛异常，有助于优化。
 * ─────────────────────────────────────────────────────────────────── */
#if !defined(__cplusplus) && GNUC_PREREQ(3, 3)
#  define MY_THROW  __attribute__((__nothrow__))
#elif defined(__cplusplus) && GNUC_PREREQ(2, 8)
#  define MY_THROW  throw()
#else
#  define MY_THROW  /* 不支持 */
#endif

#if GNUC_PREREQ(3, 2) || CLANG_PREREQ(1, 0)
#  define MY_ALWAYS_INLINE  __attribute__((__always_inline__)) static inline
#else
#  define MY_ALWAYS_INLINE  static inline
#endif

 * glibc sys/cdefs.h 原始定义（完全照录）：
 *   #ifdef __cplusplus
 *   # define __BEGIN_DECLS  extern "C" {
 *   # define __END_DECLS    }
 *   #else
 *   # define __BEGIN_DECLS
 *   # define __END_DECLS
 *   #endif
 * 任何对外头文件都应使用此对，让 C++ 可以直接包含 C 库头文件。
 * ─────────────────────────────────────────────────────────────────── */
#ifdef __cplusplus
#  define BEGIN_DECLS  extern "C" {
#  define END_DECLS    }
#else
#  define BEGIN_DECLS
#  define END_DECLS
