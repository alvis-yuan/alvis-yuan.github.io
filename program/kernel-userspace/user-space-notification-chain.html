第 1 章

Linux 内核通知链
移植到应用层

Linux 内核通知链（Notifier Chain）是一种轻量级的发布–订阅机制， 用于内核子系统之间异步事件通知。本文基于官方内核源码 (include/linux/notifier.h、 kernel/notifier.c) 将其完整移植到用户空间，实现 user_notifier.h / user_notifier.c， 并详细分析在多线程与多进程两种场景下的适用性。

原始作者与设计背景

内核通知链由 Alan Cox <Alan.Cox@linux.org> 设计， 原始注释写道： 

/*
 * Routines to manage notifier chains for passing status changes to any
 * interested routines. We need this instead of hard coded call lists so
 * that modules can poke their nose into the innards. The network devices
 * needed them so here they are for the rest of you.
 *
 *                      Alan Cox <Alan.Cox@linux.org>
 */

核心思想：避免硬编码调用列表，让内核模块能动态注册对特定事件的兴趣， 当事件发生时，按优先级依次调用所有已注册回调。

第 2 章

内核原始数据结构

以下代码直接摘录自官方源码 include/linux/notifier.h (v6.19.6)

notifier_block — 核心节点

typedef int (*notifier_fn_t)(struct notifier_block *nb,
                unsigned long action, void *data);

struct notifier_block {
    notifier_fn_t               notifier_call;  /* 回调函数指针 */
    struct notifier_block __rcu *next;          /* 下一个节点（链表） */
    int                          priority;       /* 优先级，越大越先调用 */
};
🔑 三个参数的含义
nb：指向当前 notifier_block 自身，允许回调访问扩展字段（通过 container_of）
action：事件类型码（unsigned long），一条链可以承载多种事件
data：事件附带的私有数据指针，回调自行解析

四种链头结构

/* 原子链头：spinlock + RCU 保护 */
struct atomic_notifier_head {
    spinlock_t                   lock;
    struct notifier_block __rcu *head;
};

/* 阻塞链头：读写信号量保护 */
struct blocking_notifier_head {
    struct rw_semaphore          rwsem;
    struct notifier_block __rcu *head;
};

/* 原始链头：无内置保护，调用者自行负责 */
struct raw_notifier_head {
    struct notifier_block __rcu *head;
};

/* SRCU 链头：可睡眠 RCU，适合高频调用 */
struct srcu_notifier_head {
    struct mutex                 mutex;
    struct srcu_usage             srcuu;
    struct srcu_struct            srcu;
    struct notifier_block __rcu *head;
};

静态初始化宏

#define ATOMIC_INIT_NOTIFIER_HEAD(name) do {   \
        spin_lock_init(&(name)->lock);            \
        (name)->head = NULL;                      \
    } while (0)

#define BLOCKING_INIT_NOTIFIER_HEAD(name) do { \

    } while (0)

#define RAW_INIT_NOTIFIER_HEAD(name) do {       \
        (name)->head = NULL;                      \
    } while (0)
第 3 章

四种通知链类型详解

来源： include/linux/notifier.h 注释 

/*
 * Notifier chains are of four types:
 *
 *  Atomic notifier chains: Chain callbacks run in interrupt/atomic
 *      context. Callouts are not allowed to block.
 *  Blocking notifier chains: Chain callbacks run in process context.
 *      Callouts are allowed to block.
 *  Raw notifier chains: There are no restrictions on callbacks,
 *      registration, or unregistration. All locking and protection
 *      must be provided by the caller.
 *  SRCU notifier chains: A variant of blocking notifier chains, with
 *      the same restrictions.
 *
 * atomic_notifier_chain_register() may be called from an atomic context,
 * but blocking_notifier_chain_register() and srcu_notifier_chain_register()
 * must be called from a process context. Ditto for the corresponding
 * _unregister() routines.
 *
 * atomic_notifier_chain_unregister(), blocking_notifier_chain_unregister(),
 * and srcu_notifier_chain_unregister() _must not_ be called from within
 * the call chain.
 *
 * SRCU notifier chains are an alternative form of blocking notifier chains.
 * They use SRCU (Sleepable Read-Copy Update) instead of rw-semaphores for
 * protection of the chain links. This means there is _very_ low overhead
 * in srcu_notifier_call_chain(): no cache bounces and no memory barriers.
 * As compensation, srcu_notifier_chain_unregister() is rather expensive.
 * SRCU notifier chains should be used when the chain will be called very
 * often but notifier_blocks will seldom be removed.
 */

返回值与控制流

来源： include/linux/notifier.h 

#define NOTIFY_DONE         0x0000  /* Don't care */
#define NOTIFY_OK           0x0001  /* Suits me */
#define NOTIFY_STOP_MASK    0x8000  /* Don't call further */
#define NOTIFY_BAD          (NOTIFY_STOP_MASK|0x0002)
                                    /* Bad/Veto action */
/*
 * Clean way to return from the notifier and stop further calls.
 */
#define NOTIFY_STOP         (NOTIFY_OK|NOTIFY_STOP_MASK)
宏值含义是否继续链
NOTIFY_DONE0x0000订阅者对此事件不感兴趣（无关）✅ 继续
NOTIFY_OK0x0001事件被正确处理✅ 继续
NOTIFY_STOP0x8001处理成功，但不再通知后续节点❌ 停止
NOTIFY_BAD0x8002处理出错 / 否决，停止链❌ 停止

robust 版本 — 回滚机制

/**
 * notifier_call_chain_robust - Inform the registered notifiers about an event
 *                              and rollback on error.
 * @nl:      Pointer to head of the blocking notifier chain
 * @val_up:  Value passed unmodified to the notifier function
 * @val_down: Value passed unmodified to the notifier function when recovering
 *             from an error on @val_up
 *
 * NOTE: It is important the @nl chain doesn't change between the two
 *       invocations of notifier_call_chain() such that we visit the
 *       exact same notifier callbacks; this rules out any RCU usage.
 *
 * Return: the return value of the @val_up call.
 */
static int notifier_call_chain_robust(struct notifier_block **nl,
                    unsigned long val_up, unsigned long val_down,
                    void *v)
{
    int ret, nr = 0;

    ret = notifier_call_chain(nl, val_up, v, -1, &nr);
    if (ret & NOTIFY_STOP_MASK)
        notifier_call_chain(nl, val_down, v, nr-1, NULL);

    return ret;
}
第 5 章

应用层移植设计原则

内核机制 vs 用户空间映射

内核机制用户空间替代说明
spinlockpthread_spinlock_t / pthread_mutex_t原子链注册/注销保护
rw_semaphore（rwsem）pthread_rwlock_t阻塞链读写保护
mutexpthread_mutex_tSRCU 链注册保护
RCUpthread_rwlock_t（读锁）用户空间无 RCU，用读写锁近似
__rcu 指针普通指针 + 锁用户空间无内存语义注解
EXPORT_SYMBOL_GPL普通函数，.h 声明用户空间无模块系统
图 5-1：内核机制与用户空间实现的映射关系

移植决策

  • 保留 三种链：Atomic（spinlock）、Blocking（rwlock）、Raw（无锁）；SRCU 合并入 Blocking。
  • 保留所有 NOTIFY_XXX 宏常量，与内核完全一致。
  • 保留 优先级链表插入逻辑（高优先级排前）。
  • 保留 robust 变体（val_up / val_down 回滚）。
  • 移除 RCU、kprobes、trace 等纯内核依赖，用 POSIX 线程原语替代。

user_notifier.h

完整头文件，接口与内核保持最大兼容性：

/*
 * user_notifier.h — Linux 内核通知链移植到应用层
 *
 * 严格对应 Linux 官方源码：
 *   include/linux/notifier.h  (v6.19.6)
 *   kernel/notifier.c
 *   https://elixir.bootlin.com/linux/v6.19.6/source/include/linux/notifier.h
 *   https://codebrowser.dev/linux/linux/kernel/notifier.c.html
 *
 * 内核原始注释（Alan Cox <Alan.Cox@linux.org>）：
 *   "Routines to manage notifier chains for passing status changes to any
 *    interested routines. We need this instead of hard coded call lists so
 *    that modules can poke their nose into the innards."
 *
 * 移植说明：
 *   Atomic  链 → pthread_rwlock_t 读写锁（保护注册/注销）+ 读锁（遍历）
 *   Blocking 链 → pthread_rwlock_t（与内核 rwsem 语义对应）
 *   Raw      链 → 无内置保护，调用者全权负责（对应 raw_notifier_*）
 *   SRCU     链 → 合并到 Blocking 实现（用户空间无 SRCU 原语）
 */
#ifndef USER_NOTIFIER_H
#define USER_NOTIFIER_H

#include <pthread.h>
#include <errno.h>
#include <stddef.h>
#include <stdint.h>

/* ── 返回值宏（完全对应内核 include/linux/notifier.h）─────────────────
 *
 * 原始定义来源：
 *   https://elixir.bootlin.com/linux/v6.19.6/source/include/linux/notifier.h
 *
 * #define NOTIFY_DONE       0x0000   // Don't care
 * #define NOTIFY_OK         0x0001   // Suits me
 * #define NOTIFY_STOP_MASK  0x8000   // Don't call further
 * #define NOTIFY_BAD        (NOTIFY_STOP_MASK|0x0002)  // Bad/Veto action
 * #define NOTIFY_STOP       (NOTIFY_OK|NOTIFY_STOP_MASK)
 */
#define NOTIFY_DONE         0x0000  /* 不感兴趣 */
#define NOTIFY_OK           0x0001  /* 处理成功 */
#define NOTIFY_STOP_MASK    0x8000  /* 停止标志位 */
#define NOTIFY_BAD          (NOTIFY_STOP_MASK | 0x0002)  /* 否决/错误 */
#define NOTIFY_STOP         (NOTIFY_OK | NOTIFY_STOP_MASK)  /* 成功但停止 */

/* ── notifier_block（对应内核 struct notifier_block）────────────────── */
struct user_notifier_block;

/* 回调函数类型（对应内核 notifier_fn_t）*/
typedef int (*user_notifier_fn_t)(
    struct user_notifier_block *nb,
    unsigned long               action,
    void                       *data
);

struct user_notifier_block {
    user_notifier_fn_t          notifier_call;  /* 回调函数 */
    struct user_notifier_block *next;           /* 链表下一节点 */
    int                         priority;       /* 优先级，越大越先调用 */
};

/* ── 链头结构 ───────────────────────────────────────────────────────── */

/* Atomic 链头（注册用 write-lock，遍历用 read-lock）*/
struct user_atomic_notifier_head {
    pthread_rwlock_t            lock;
    struct user_notifier_block *head;
};

/* Blocking 链头（对应内核 struct blocking_notifier_head，rwsem→rwlock）*/
struct user_blocking_notifier_head {
    pthread_rwlock_t            rwlock;
    struct user_notifier_block *head;
};

/* Raw 链头（无任何内置保护，对应内核 struct raw_notifier_head）*/
struct user_raw_notifier_head {
    struct user_notifier_block *head;
};

/* ── 初始化宏（对应内核 ATOMIC_INIT_NOTIFIER_HEAD 等）────────────────── */
#define USER_ATOMIC_INIT_NOTIFIER_HEAD(nh)   \
    do {                                     \
        pthread_rwlock_init(&(nh)->lock, NULL); \
        (nh)->head = NULL;                   \
    } while (0)

#define USER_BLOCKING_INIT_NOTIFIER_HEAD(nh) \
    do {                                     \
        pthread_rwlock_init(&(nh)->rwlock, NULL); \
        (nh)->head = NULL;                   \
    } while (0)

#define USER_RAW_INIT_NOTIFIER_HEAD(nh)     \
    do { (nh)->head = NULL; } while (0)

/* ── 销毁宏 ─────────────────────────────────────────────────────────── */
#define USER_ATOMIC_DESTROY_NOTIFIER_HEAD(nh)   \
    pthread_rwlock_destroy(&(nh)->lock)
#define USER_BLOCKING_DESTROY_NOTIFIER_HEAD(nh) \
    pthread_rwlock_destroy(&(nh)->rwlock)

/* ── Atomic 链 API ──────────────────────────────────────────────────── */
int user_atomic_notifier_chain_register(
    struct user_atomic_notifier_head *nh,
    struct user_notifier_block       *n);

int user_atomic_notifier_chain_unregister(
    struct user_atomic_notifier_head *nh,
    struct user_notifier_block       *n);

int user_atomic_notifier_call_chain(

    struct user_blocking_notifier_head *nh,
    struct user_notifier_block         *n);

int user_blocking_notifier_chain_unregister(
    struct user_blocking_notifier_head *nh,
    struct user_notifier_block         *n);

int user_blocking_notifier_call_chain(
    struct user_blocking_notifier_head *nh,
    unsigned long                       val,
    void                               *v);

int user_blocking_notifier_call_chain_robust(
    struct user_blocking_notifier_head *nh,
    unsigned long                       val_up,
    unsigned long                       val_down,
    void                               *v);

user_notifier.c

完整实现文件，逻辑与内核 kernel/notifier.c 严格对应：

/*
 * user_notifier.c — Linux 内核通知链移植到应用层
 *
 * 对应内核源码：kernel/notifier.c
 * 参考：https://codebrowser.dev/linux/linux/kernel/notifier.c.html
 *
 * 编译：gcc -o demo demo.c user_notifier.c -lpthread
 */
#include "user_notifier.h"
#include <errno.h>
#include <stddef.h>

/* ═══════════════════════════════════════════════════════════════════════
 * 内部核心函数（对应内核 static 函数）
 * ═══════════════════════════════════════════════════════════════════════ */

/**
 * notifier_chain_register_internal - 按优先级插入链表
 *
 * 对应内核：
 *   static int notifier_chain_register(struct notifier_block **nl,
 *                                      struct notifier_block *n,
 *                                      bool unique_priority)
 *   kernel/notifier.c:17-37
 *
 * 内核原始逻辑：
 *   while ((*nl) != NULL) {
 *       if (n->priority > (*nl)->priority)
 *           break;
 *       nl = &((*nl)->next);
 *   }
 *   n->next = *nl;
 *   rcu_assign_pointer(*nl, n);
 *   return 0;
 *
 * 用户空间：去掉 RCU，直接指针赋值（受外部锁保护）
 */
static int notifier_chain_register_internal(
    struct user_notifier_block **nl,
    struct user_notifier_block  *n)
{
    while (*nl != NULL) {
        if (*nl == n)           /* 已注册（对应内核 WARN -EEXIST）*/
            return -EEXIST;
        if (n->priority > (*nl)->priority)
            break;              /* 找到插入位置 */
        nl = &((*nl)->next);
    }
    n->next = *nl;
    *nl = n;                    /* 替代 rcu_assign_pointer */
    return 0;
}

/**
 * notifier_chain_unregister_internal - 从链表移除节点
 *
 * 对应内核：
 *   static int notifier_chain_unregister(struct notifier_block **nl,
 *                                        struct notifier_block *n)
 *   kernel/notifier.c:39-51
 *
 * 返回 0 成功，-ENOENT 未找到
 */
static int notifier_chain_unregister_internal(
    struct user_notifier_block **nl,
    struct user_notifier_block  *n)
{
    while (*nl != NULL) {
        if (*nl == n) {
            *nl = n->next;      /* 替代 rcu_assign_pointer */
            return 0;
        }
        nl = &((*nl)->next);
    }
    return -ENOENT;
}

/**
 * notifier_call_chain_internal - 遍历通知链，依次调用回调
 *
 * 对应内核：
 *   static int notifier_call_chain(struct notifier_block **nl,
 *                                  unsigned long val, void *v,
 *                                  int nr_to_call, int *nr_calls)
 *   kernel/notifier.c:65-96
 *
 * 核心逻辑（内核原文）：
 *   ret = nb->notifier_call(nb, val, v);
 *   if (ret & NOTIFY_STOP_MASK)
 *       break;
 *
 * @nr_to_call: 最多调用数，-1 表示全部
 * @nr_calls:   记录实际调用次数，NULL 忽略
 * @return:     最后一个被调用的回调的返回值
 */
static int notifier_call_chain_internal(
    struct user_notifier_block **nl,
    unsigned long                val,
    void                        *v,
    int                          nr_to_call,
    int                         *nr_calls)
{
    int ret = NOTIFY_DONE;
    struct user_notifier_block *nb, *next_nb;

    nb = *nl;

    while (nb && nr_to_call) {
        next_nb = nb->next;         /* 替代 rcu_dereference_raw */

        ret = nb->notifier_call(nb, val, v);

        if (nr_calls)
            (*nr_calls)++;

        if (ret & NOTIFY_STOP_MASK)  /* 内核原文条件完全一致 */
            break;

        nb = next_nb;
        nr_to_call--;
    }
    return ret;
}

/**
 * notifier_call_chain_robust_internal - 带回滚的通知链调用
 *
 * 对应内核：
 *   static int notifier_call_chain_robust(struct notifier_block **nl,
 *                                         unsigned long val_up,
 *                                         unsigned long val_down,
 *                                         void *v)
 *   kernel/notifier.c:114-125
 *
 * 内核注释原文：
 *   "NOTE: It is important the @nl chain doesn't change between the two
 *    invocations of notifier_call_chain() such that we visit the
 *    exact same notifier callbacks; this rules out any RCU usage."
 */
static int notifier_call_chain_robust_internal(
    struct user_notifier_block **nl,
    unsigned long                val_up,
    unsigned long                val_down,
    void                        *v)
{
    int ret, nr = 0;

    ret = notifier_call_chain_internal(nl, val_up, v, -1, &nr);
    if (ret & NOTIFY_STOP_MASK)
        notifier_call_chain_internal(nl, val_down, v, nr - 1, NULL);
    return ret;
}

/* ═══════════════════════════════════════════════════════════════════════
 * Atomic 链（对应内核 atomic_notifier_chain_*）
 * 注册/注销：write-lock；遍历：read-lock
 * ═══════════════════════════════════════════════════════════════════════ */

/**
 * user_atomic_notifier_chain_register
 *
 * 对应内核：
 *   int atomic_notifier_chain_register(struct atomic_notifier_head *nh,
 *                                      struct notifier_block *n)
 *   kernel/notifier.c:141-151
 *
 * 内核用 spin_lock_irqsave，用户空间用 pthread_rwlock_wrlock
 * Returns 0 on success, -EEXIST if already registered
 */
int user_atomic_notifier_chain_register(
    struct user_atomic_notifier_head *nh,
    struct user_notifier_block       *n)
{
    int ret;
    pthread_rwlock_wrlock(&nh->lock);
    ret = notifier_chain_register_internal(&nh->head, n);
    pthread_rwlock_unlock(&nh->lock);
    return ret;
}

/**
 * user_atomic_notifier_chain_unregister
 *
 * 对应内核：
 *   int atomic_notifier_chain_unregister(struct atomic_notifier_head *nh,
 *                                        struct notifier_block *n)
 *   kernel/notifier.c:186-198
 *
 * 内核调用 synchronize_rcu()，用户空间无需此步（无 RCU 读者）
 * Returns 0 on success, -ENOENT on failure
 */
int user_atomic_notifier_chain_unregister(
    struct user_atomic_notifier_head *nh,
    struct user_notifier_block       *n)
{
    int ret;
    pthread_rwlock_wrlock(&nh->lock);
    ret = notifier_chain_unregister_internal(&nh->head, n);
    pthread_rwlock_unlock(&nh->lock);
    return ret;
}

/**
 * user_atomic_notifier_call_chain
 *
 * 对应内核：
 *   int atomic_notifier_call_chain(struct atomic_notifier_head *nh,
 *                                  unsigned long val, void *v)
 *   kernel/notifier.c:217-228
 *
 * 内核用 rcu_read_lock/unlock，用户空间用 read-lock（允许并发读）
 */
int user_atomic_notifier_call_chain(
    struct user_atomic_notifier_head *nh,
    unsigned long                     val,
    void                             *v)
{
    int ret;
    pthread_rwlock_rdlock(&nh->lock);
    ret = notifier_call_chain_internal(&nh->head, val, v, -1, NULL);
    pthread_rwlock_unlock(&nh->lock);
    return ret;
}

/* ═══════════════════════════════════════════════════════════════════════
 * Blocking 链（对应内核 blocking_notifier_chain_*）
 * 注册/注销：write-lock（rwlock）；遍历：read-lock
 * ═══════════════════════════════════════════════════════════════════════ */

/**
 * user_blocking_notifier_chain_register
 *
 * 对应内核：
 *   int blocking_notifier_chain_register(struct blocking_notifier_head *nh,
 *                                        struct notifier_block *n)
 *   kernel/notifier.c:279-283
 *
 * 内核用 down_write(&nh->rwsem)，用户空间对应 pthread_rwlock_wrlock
 */
int user_blocking_notifier_chain_register(
    struct user_blocking_notifier_head *nh,
    struct user_notifier_block         *n)
{
    int ret;
    pthread_rwlock_wrlock(&nh->rwlock);
    ret = notifier_chain_register_internal(&nh->head, n);
    pthread_rwlock_unlock(&nh->rwlock);
    return ret;
}

/**
 * user_blocking_notifier_chain_unregister
 *
 * 对应内核：
 *   int blocking_notifier_chain_unregister(struct blocking_notifier_head *nh,
 *                                          struct notifier_block *n)
 *   kernel/notifier.c:313-330
 *
 * 内核用 down_write(&nh->rwsem)，用户空间对应 pthread_rwlock_wrlock
 */
int user_blocking_notifier_chain_unregister(
    struct user_blocking_notifier_head *nh,
    struct user_notifier_block         *n)
{
    int ret;
    pthread_rwlock_wrlock(&nh->rwlock);
    ret = notifier_chain_unregister_internal(&nh->head, n);
    pthread_rwlock_unlock(&nh->rwlock);
    return ret;
}

/**
 * user_blocking_notifier_call_chain
 *
 * 对应内核：
 *   int blocking_notifier_call_chain(struct blocking_notifier_head *nh,
 *                                    unsigned long val, void *v)
 *   kernel/notifier.c:368-384
 *
 * 内核注释：
 *   "Calls each function in a notifier chain in turn. The functions
 *    run in a process context, so they are allowed to block."
 *
 * 内核用 down_read(&nh->rwsem)，用户空间对应 pthread_rwlock_rdlock
 */
int user_blocking_notifier_call_chain(
    struct user_blocking_notifier_head *nh,
    unsigned long                       val,
    void                               *v)
{
    int ret = NOTIFY_DONE;
    if (nh->head) {                 /* 对应内核 rcu_access_pointer 检查 */
        pthread_rwlock_rdlock(&nh->rwlock);
        ret = notifier_call_chain_internal(&nh->head, val, v, -1, NULL);
        pthread_rwlock_unlock(&nh->rwlock);
    }
    return ret;
}

/**
 * user_blocking_notifier_call_chain_robust
 *
 * 对应内核：
 *   int blocking_notifier_call_chain_robust(
 *       struct blocking_notifier_head *nh,
 *       unsigned long val_up, unsigned long val_down, void *v)
 *   kernel/notifier.c:333-350
 */
int user_blocking_notifier_call_chain_robust(
    struct user_blocking_notifier_head *nh,
    unsigned long                       val_up,
    unsigned long                       val_down,
    void                               *v)
{
    int ret = NOTIFY_DONE;
    if (nh->head) {
        pthread_rwlock_rdlock(&nh->rwlock);
        ret = notifier_call_chain_robust_internal(
            &nh->head, val_up, val_down, v);
        pthread_rwlock_unlock(&nh->rwlock);
    }
    return ret;
}

/* ═══════════════════════════════════════════════════════════════════════
 * Raw 链（对应内核 raw_notifier_chain_*）
 * 完全无内置保护，调用者全权负责互斥
 * ═══════════════════════════════════════════════════════════════════════ */

/**
 * user_raw_notifier_chain_register
 *
 * 对应内核：
 *   int raw_notifier_chain_register(struct raw_notifier_head *nh,
 *                                   struct notifier_block *n)
 *   kernel/notifier.c:402-406
 *
 * 内核注释：
 *   "Adds a notifier to a raw notifier chain.
 *    All locking must be provided by the caller."
 */
int user_raw_notifier_chain_register(
    struct user_raw_notifier_head *nh,
    struct user_notifier_block    *n)
{
    return notifier_chain_register_internal(&nh->head, n);
}

/**
 * user_raw_notifier_chain_unregister
 *
 * 对应内核：
 *   int raw_notifier_chain_unregister(struct raw_notifier_head *nh,
 *                                     struct notifier_block *n)
 *   kernel/notifier.c:419-423
 */
int user_raw_notifier_chain_unregister(
    struct user_raw_notifier_head *nh,
    struct user_notifier_block    *n)
{
    return notifier_chain_unregister_internal(&nh->head, n);
}

/**
 * user_raw_notifier_call_chain
 *
 * 对应内核：
 *   int raw_notifier_call_chain(struct raw_notifier_head *nh,
 *                               unsigned long val, void *v)
 *   kernel/notifier.c:450-454
 *
 * 内核注释：
 *   "Calls each function in a notifier chain in turn. The functions
 *    run in an undefined context.
 *    All locking must be provided by the caller."
 */
int user_raw_notifier_call_chain(
    struct user_raw_notifier_head *nh,
    unsigned long                  val,
    void                          *v)
{
    return notifier_call_chain_internal(&nh->head, val, v, -1, NULL);
}

int user_raw_notifier_call_chain_robust(
    struct user_raw_notifier_head *nh,
    unsigned long                  val_up,
    unsigned long                  val_down,

{
    return notifier_call_chain_robust_internal(
        &nh->head, val_up, val_down, v);
}
第 8 章

使用示例

示例 1：多线程阻塞链（类比内核 reboot_notifier）

#include "user_notifier.h"
#include <stdio.h>
#include <pthread.h>

/* 事件类型定义（类比内核 sys_down 事件码）*/
#define EVT_SHUTDOWN  0x01
#define EVT_CONFIG    0x02

/* 全局阻塞链（对应内核 BLOCKING_NOTIFIER_HEAD(reboot_notifier_list)）*/
static struct user_blocking_notifier_head my_chain;

/* 回调函数 A（高优先级）*/
static int handler_a(struct user_notifier_block *nb,
                     unsigned long action, void *data)
{
    if (action == EVT_SHUTDOWN)
        printf("[A] 收到 SHUTDOWN 事件，执行清理...\n");
    return NOTIFY_OK;
}

/* 回调函数 B（低优先级），否决某事件 */
static int handler_b(struct user_notifier_block *nb,
                     unsigned long action, void *data)
{
    if (action == EVT_CONFIG) {
        printf("[B] 拒绝配置变更！\n");
        return NOTIFY_BAD;   /* 停止链并否决 */
    }
    return NOTIFY_DONE;
}

/* notifier_block 实例（对应内核 static struct notifier_block xxx）*/
static struct user_notifier_block nb_a = {
    .notifier_call = handler_a,
    .priority      = 100,      /* 高优先级，先调用 */
};
static struct user_notifier_block nb_b = {
    .notifier_call = handler_b,
    .priority      = 50,
};

void *worker(void *arg) {
    int ret = user_blocking_notifier_call_chain(&my_chain, EVT_SHUTDOWN, NULL);
    printf("链调用结果: 0x%04x\n", ret);
    return NULL;
}

int main(void) {
    USER_BLOCKING_INIT_NOTIFIER_HEAD(&my_chain);

    /* 注册（对应内核 blocking_notifier_chain_register）*/
    user_blocking_notifier_chain_register(&my_chain, &nb_a);
    user_blocking_notifier_chain_register(&my_chain, &nb_b);

    pthread_t tid;
    pthread_create(&tid, NULL, worker, NULL);
    pthread_join(tid, NULL);

    /* 注销 */
    user_blocking_notifier_chain_unregister(&my_chain, &nb_a);
    user_blocking_notifier_chain_unregister(&my_chain, &nb_b);

    USER_BLOCKING_DESTROY_NOTIFIER_HEAD(&my_chain);
    return 0;
}

示例 2：robust 回滚（类比内核 blocking_notifier_call_chain_robust）

#define EVT_PREPARE   1   /* val_up   */
#define EVT_ROLLBACK  2   /* val_down */

static int prepare_handler(struct user_notifier_block *nb,
                           unsigned long action, void *data)
{
    if (action == EVT_PREPARE) {
        printf("[%s] 执行 PREPARE\n", (char *)nb);
        /* 模拟某个处理器拒绝 */
        if (strcmp((char *)nb, "C") == 0)
            return NOTIFY_BAD;
    } else if (action == EVT_ROLLBACK) {
        printf("[%s] 回滚！\n", (char *)nb);
    }
    return NOTIFY_OK;
}

/* robust 调用：C 失败 → 自动对已调用的 A、B 发 ROLLBACK */
int ret = user_blocking_notifier_call_chain_robust(
    &my_chain, EVT_PREPARE, EVT_ROLLBACK, NULL);
第 9 章

多线程 vs 多进程：适用场景分析

📌 结论先行
user_notifier 天然适合多线程场景，因为链表和锁存储在进程的共享内存空间中； 多进程场景需要额外的 IPC 机制（共享内存 + 进程间锁）才能使用，复杂度大幅提升。
图 9-1：多线程与多进程的内存模型差异

✅ 多线程优势

  • 链表、锁、回调指针天然共享
  • 无需任何 IPC 机制
  • pthread_rwlock 即可直接使用
  • 回调函数在同一地址空间直接执行
  • 零拷贝，延迟极低（纳秒级）

❌ 多进程限制

  • 各进程有独立地址空间，指针不通用
  • 函数指针无法跨进程传递
  • 需共享内存 + PTHREAD_PROCESS_SHARED 锁
  • 回调无法直接执行，需消息转发
  • 注销时节点释放复杂（谁管理内存？）
第 10 章

多线程适用场景（推荐）

以下均为 user_notifier 在多线程环境中的天然适用场景：

🟢 场景 1：插件/模块热插拔事件通知

推荐 Blocking 链 多线程

主线程加载/卸载插件后，通过通知链广播给所有功能子模块。 插件注册自己的 user_notifier_block，主模块调用 user_blocking_notifier_call_chain。

类比内核场景：网络子系统通过 register_netdevice_notifier 监听网卡插拔事件。

/* 事件码 */
#define PLUGIN_LOAD    1
#define PLUGIN_UNLOAD  2

struct plugin_event {
    const char *name;
    void       *handle;
};

static int security_plugin_notifier(struct user_notifier_block *nb,
                                    unsigned long action, void *data)
{
    struct plugin_event *ev = data;
    if (action == PLUGIN_LOAD)
        printf("安全模块: 插件 %s 已加载\n", ev->name);
    return NOTIFY_OK;
}

/* 注册（由安全子模块在初始化线程中调用）*/
static struct user_notifier_block sec_nb = {
    .notifier_call = security_plugin_notifier,
    .priority      = 200,
};
user_blocking_notifier_chain_register(&plugin_chain, &sec_nb);

/* 主线程发布事件 */
struct plugin_event ev = { .name = "libfoo.so" };
user_blocking_notifier_call_chain(&plugin_chain, PLUGIN_LOAD, &ev);

🟢 场景 2：状态机转换广播

推荐 Blocking 链 多线程

应用程序有多个状态（初始化 → 运行 → 关闭），状态转换时需通知多个子线程（网络线程、UI线程、存储线程）分别做好准备。 Blocking 链天然支持回调阻塞等待，保证每个子系统都完成准备。

类比内核场景：register_reboot_notifier 在系统关机时通知各驱动清理资源。

#define STATE_RUNNING  1
#define STATE_STOPPING 2

/* 网络线程监听 */
static int network_state_handler(struct user_notifier_block *nb,
                                 unsigned long action, void *data)
{
    if (action == STATE_STOPPING) {
        printf("网络线程: 关闭所有连接...\n");
        close_all_connections();   /* 可以阻塞！Blocking 链允许 */
    }
    return NOTIFY_OK;
}

🟢 场景 3：配置变更通知（含否决）

推荐 robust 变体 多线程

配置变更时，先通过 val_up（PREPARE）征询各模块意见；若某模块返回 NOTIFY_BAD， 自动以 val_down（ROLLBACK）回滚已通知的模块。完美对应内核 robust 机制。 

#define CFG_PREPARE   1
#define CFG_COMMIT    2
#define CFG_ROLLBACK  3

int ret = user_blocking_notifier_call_chain_robust(
    &config_chain,
    CFG_PREPARE,   /* val_up：逐个通知 */
    CFG_ROLLBACK,  /* val_down：失败时回滚 */
    &new_config
);
if (!(ret & NOTIFY_STOP_MASK))
    user_blocking_notifier_call_chain(&config_chain, CFG_COMMIT, &new_config);

🟢 场景 4：高频信号/传感器事件分发

推荐 Raw 链（调用者加锁） 多线程

传感器数据高频到达（如 1kHz），每次到达触发多个处理器（滤波器、日志、告警）。 使用 Raw 链，在外层一次性加写锁（注册期间）、不加锁（读期间）， 获得最高吞吐。适合注册后极少变化、只频繁通知的场景。 类比内核 SRCU 链的设计哲学。 

/* 注册阶段（启动时，单线程）*/
user_raw_notifier_chain_register(&sensor_chain, &filter_nb);
user_raw_notifier_chain_register(&sensor_chain, &log_nb);
user_raw_notifier_chain_register(&sensor_chain, &alert_nb);

/* 高频调用阶段（无锁，极低延迟）*/
void on_sensor_data(float value) {
    /* 注册完成后链不再变化，无需加锁 */
    user_raw_notifier_call_chain(&sensor_chain, EVT_DATA, &value);
}
第 11 章

多进程场景分析

⚠️ 重要限制
user_notifier 的核心数据结构（链表节点、函数指针）存储在进程私有地址空间， 无法直接跨进程共享。在多进程场景下，通知链只能作为进程内部的分发机制， 进程间通知需借助 IPC 桥接。

多进程的两种使用模式

🟡 模式 A：进程内链 + IPC 桥接（可行）

适合 多进程

每个进程内部维护自己的 user_notifier 链，通过 Unix Domain Socket / 管道 / 消息队列 接收来自其他进程的事件，再本地分发给链上各订阅者。

图 11-1：多进程模式 A — 进程内 notifier + IPC 桥接

🟡 模式 B：共享内存 + 进程间锁（高复杂度）

复杂度高 多进程

将链头和节点放在 mmap 共享内存中，使用 PTHREAD_PROCESS_SHARED 属性的 pthread_rwlock_t。 但函数指针仍然无法跨进程使用（ASLR、不同代码段）， 回调必须用事件ID替代，接收方本地再分发。 

/* 共享内存中的结构 */
struct shared_notifier_head {
    pthread_rwlock_t lock;   /* PTHREAD_PROCESS_SHARED */
    uint32_t         event;  /* 最新事件（无函数指针！）*/
    uint64_t         data;   /* 序列化数据 */
};

/* 初始化（生产者进程）*/
pthread_rwlockattr_t attr;
pthread_rwlockattr_init(&attr);
pthread_rwlockattr_setpshared(&attr, PTHREAD_PROCESS_SHARED);
pthread_rwlock_init(&shm->lock, &attr);

/* 消费者进程：本地 notifier 链仍然有效（本进程内）*/
/* 但回调需要本地注册，不能从共享内存获取函数指针 */

多进程适合的具体用例

🟡 场景 5：守护进程接收 Unix Signal 后本地分发

适合（进程内 Raw/Blocking 链） 多进程触发，本地分发

守护进程（如 nginx worker）收到外部信号（SIGHUP / SIGUSR1）后， 在信号处理函数中设置标志位，由主循环调用 user_blocking_notifier_call_chain 通知进程内所有子模块（日志轮转、配置重载、统计清空）。 

static volatile sig_atomic_t reload_pending = 0;
static struct user_blocking_notifier_head daemon_chain;

void sig_hup_handler(int sig) {
    reload_pending = 1;   /* 信号安全：仅设标志 */
}

/* 主循环 */
while (running) {
    if (reload_pending) {
        reload_pending = 0;
        /* 通知进程内所有模块（阻塞，允许 I/O 操作）*/
        user_blocking_notifier_call_chain(
            &daemon_chain, EVT_RELOAD_CONFIG, NULL);
    }
    do_work();
}

🟡 场景 6：父子进程 fork 后各自独立的 notifier

每进程独立链 适合 fork 模型

服务器 fork 出 worker 进程，每个 worker 有自己的 user_notifier 链。 父进程通过 Unix Domain Socket 广播消息，每个 worker 的接收线程调用本地 notifier 链分发事件。 链完全独立，互不干扰，注销也完全安全。 

/* worker 进程中 */
void worker_main(int sock_fd) {
    struct user_blocking_notifier_head local_chain;
    USER_BLOCKING_INIT_NOTIFIER_HEAD(&local_chain);

    /* 注册本 worker 自己的处理器 */
    user_blocking_notifier_chain_register(&local_chain, &my_handler_nb);

    /* IPC 接收线程将外部事件桥接到本地链 */
    start_ipc_bridge_thread(sock_fd, &local_chain);

    run_event_loop();
}

总结

✅ 最终建议
多线程场景：直接使用 user_notifier，选择合适链类型：
  • 高频无阻塞 → Raw 链（注册期加锁，运行期无锁）
  • 通用阻塞回调 → Blocking 链（pthread_rwlock，读多写少）
  • 无阻塞并发 → Atomic 链（读写锁保护，回调不可阻塞）

多进程场景：在每个进程内部使用 user_notifier（Blocking 链）， 进程间事件通过 IPC（Unix Socket / 管道 / 消息队列）桥接传入， 由本地接收线程调用 user_blocking_notifier_call_chain 进行进程内分发。

核心原则：通知链是进程内的分发机制，IPC 是进程间的传输机制， 两者分工明确，组合使用效果最佳。
