GLib 设计哲学概述

GLib 是 GNOME 项目的基础 C 工具库，涵盖数据结构、内存管理、线程、事件循环、错误处理等模块。即使不直接引入 GLib 作为依赖，其设计思想同样适用于日常 C 系统编程。

RAII — 自动资源管理

GLib 从 2.44 版本起通过 g_autoptr()、g_autofree、g_auto() 宏将 GCC/Clang 的 __attribute__((cleanup)) 扩展封装为易用的 RAII 工具，使 C 代码在变量离开作用域时自动释放资源，彻底消除"多出口提前返回时忘记 free"的内存泄漏问题。

📎 官方文档来源

docs.gtk.org/glib/auto-cleanup.html — GLib: Automatic Cleanup（GLib 2.44+）

三个核心宏

宏适用场景底层调用引入版本
g_autofree简单堆内存指针（malloc 分配）始终调用 g_free()2.44
g_autoptr(T)指针类型，有注册的 cleanup 函数调用 T 对应的 cleanup 函数2.44
g_auto(T)栈上分配的结构体（非指针）调用 T 对应的 clear 函数2.44

GLib 官方示例（g_autoptr）

以下示例直接引自 GLib 官方文档：

gboolean
check_exists(GVariant *dict)
{
  g_autoptr(GVariant) dirname, basename = NULL;
  g_autofree gchar *path = NULL;

  dirname  = g_variant_lookup_value(dict, "dirname",  G_VARIANT_TYPE_STRING);
  if (dirname == NULL)
    return FALSE;  /* dirname 自动 unref */

  basename = g_variant_lookup_value(dict, "basename", G_VARIANT_TYPE_STRING);
  if (basename == NULL)
    return FALSE;  /* dirname + basename 均自动 unref */

  path = g_build_filename(g_variant_get_string(dirname, NULL),
                           g_variant_get_string(basename, NULL),
                           NULL);

  return g_access(path, R_OK) == 0;
  /* path / dirname / basename 在此全部自动释放 */
}

在自己代码中模拟（无需引入 GLib）

✅ 原理

g_autofree 的底层实现仅用到 GCC/Clang 的 __attribute__((cleanup(func)))，可在不引入 GLib 的情况下直接仿写。

/* ── 在自己项目中实现 RAII 工具宏 ── */
#include <stdlib.h>
#include <stdio.h>

/* 释放普通堆内存 */
static inline void cleanup_free(void *p) {
    void **pp = (void **)p;
    if (*pp) { free(*pp); *pp = NULL; }
}

/* 关闭文件描述符 */
static inline void cleanup_fd(int *fd) {
    if (*fd >= 0) { close(*fd); *fd = -1; }
}

/* 关闭 FILE* */
static inline void cleanup_file(FILE **fp) {
    if (*fp) { fclose(*fp); *fp = NULL; }
}

#define autofree  __attribute__((cleanup(cleanup_free)))
#define autofd    __attribute__((cleanup(cleanup_fd)))
#define autofile  __attribute__((cleanup(cleanup_file)))

/* ── 使用示例 ── */
int process_file(const char *path) {
    autofd  int   fd   = open(path, O_RDONLY);
    autofree char *buf  = malloc(4096);

    if (fd < 0 || !buf) return -1; /* 自动清理 */

    ssize_t n = read(fd, buf, 4096);
    if (n < 0) return -1;         /* 自动清理 */

    printf("Read %zd bytes\n", n);
    return 0; /* fd 和 buf 在此自动清理，逆序执行 */
}
❌ 传统写法（易泄漏）
char *buf = malloc(4096);
int fd = open(path, ...);
if (fd < 0) {
  free(buf); // 容易忘
  return -1;
}
// ...更多 goto cleanup 样板代码
close(fd);
free(buf);
✅ RAII 写法（自动清理）
autofree char *buf = malloc(4096);
autofd   int   fd  = open(path, ...);
if (fd < 0) return -1;
// 无需手动清理
// 无论哪里 return，自动逆序清理
⚠️ 注意事项（来自官方文档）
  • 必须在声明时立即初始化变量（如 = NULL 或 = -1）
  • 不要通过 goto 跳过声明行（cleanup 可能操作未初始化的值）
  • 不能用于类成员变量，仅限于函数局部变量
  • 仅支持 GCC 和 Clang，MSVC 不支持此扩展

引用计数 — RcBox 与 grefcount

GLib 提供 grefcount / gatomicrefcount 基础类型，以及更高级的 RcBox（Reference Counted Box）封装，让普通 C 结构体能够享有引用计数式的内存管理，无需引入 GObject 类型系统。

📎 官方文档来源

docs.gtk.org/glib/reference-counting.html（GLib 2.58+）

引用计数生命周期图

官方示例：RcBox 基本用法

/* 官方文档示例：docs.gtk.org/glib/reference-counting.html */

typedef struct {
  char *name;
  char *address;
  char *city;
  int   age;
} Person;

Person *person_new(void) {
  return g_rc_box_new0(Person);  /* 零初始化 + 引用计数=1 */
}

void person_clear(Person *p) {
  g_free(p->name);
  g_free(p->address);
  g_free(p->city);
}

void add_person_to_database(Database *db, Person *p) {
  /* 数据库持有引用：ref_count++ */
  db->persons = g_list_prepend(db->persons, g_rc_box_acquire(p));
}

void remove_person_from_database(Database *db, Person *p) {
  db->persons = g_list_remove(db->persons, p);
  /* release 时若 ref_count 降为 0，自动调用 person_clear 再 free */
  g_rc_box_release_full(p, (GDestroyNotify) person_clear);
}

在自己代码中实现引用计数模式

/* 无需 GLib，实现最简引用计数结构 */
#include <stdatomic.h>
#include <stdlib.h>

typedef void (*DestroyFunc)(void *);

typedef struct {
  _Atomic int ref_count;
  DestroyFunc destroy;
  /* 之后紧跟用户数据 */
} RcHeader;

void *rc_new(size_t size, DestroyFunc destroy) {
  RcHeader *h = calloc(1, sizeof(RcHeader) + size);
  atomic_store(&h->ref_count, 1);
  h->destroy = destroy;
  return (char *)h + sizeof(RcHeader);
}

void *rc_acquire(void *ptr) {
  RcHeader *h = (RcHeader *)ptr - 1;
  atomic_fetch_add(&h->ref_count, 1);
  return ptr;
}

void rc_release(void *ptr) {
  RcHeader *h = (RcHeader *)ptr - 1;
  if (atomic_fetch_sub(&h->ref_count, 1) == 1) {
    if (h->destroy) h->destroy(ptr);
    free(h);
  }
}

结构化错误处理 — GError 模式

GLib 提供 GError 作为统一的、结构化的错误报告机制，解决 C 语言用数字错误码难以携带上下文信息的问题，并配套了一套严格的使用规则。

📎 官方文档来源

docs.gtk.org/glib/error-reporting.html — GLib: Error Reporting

GError 结构（官方）

/* GError 结构包含三个字段（来自官方文档）：*/
struct GError {
  GQuark  domain;   /* 错误所在模块（用 quark 字符串唯一标识）*/
  int     code;     /* 该模块内具体错误码（枚举值）*/
  char   *message;  /* 人类可读的错误描述（UTF-8）*/
};

官方示例：设置与传播错误

/* 官方文档示例：定义错误域 */
#define FOO_ERROR foo_error_quark()
GQuark foo_error_quark(void) {
  return g_quark_from_static_string("foo-error-quark");
}
typedef enum { FOO_ERROR_NOT_FOUND, FOO_ERROR_PERMISSION } FooError;

/* 底层：设置错误 */
int foo_open_file(GError **error) {
  g_return_val_if_fail(error == NULL || *error == NULL, -1);

  int fd = open("file.txt", O_RDONLY);
  if (fd < 0) {
    g_set_error(error,
               FOO_ERROR, FOO_ERROR_NOT_FOUND,
               "Failed to open file: %s", g_strerror(errno));
    return -1;
  }
  return fd;
}

/* 调用方：使用错误 */
void caller(void) {
  g_autoptr(GError) err = NULL;
  int fd = foo_open_file(&err);
  if (err) {
    if (g_error_matches(err, FOO_ERROR, FOO_ERROR_NOT_FOUND))
      fprintf(stderr, "[NOT FOUND] %s\n", err->message);
    else
      fprintf(stderr, "[ERROR] %s\n", err->message);
    /* g_autoptr 在函数结束时自动 g_error_free(err) */
  }
}

官方规则摘要（GLib 文档）

  • GError 只用于可恢复的运行时错误，编程错误用 g_assert / g_return_if_fail
  • 函数的最后一个参数若能报错，应为 GError **error
  • 调用方可传 NULL 表示忽略错误（但函数仍须中止当前操作）
  • GError* 传入前必须初始化为 NULL
  • 不可"堆叠"错误：每次错误后必须处理或向上传播
  • GError 结构体不可在栈上分配

在自己代码中实现结构化错误

/* 仿 GError 的轻量错误结构，无需 GLib */
#include <stdarg.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
  int   domain;   /* 模块 ID，如 ERR_DOMAIN_IO */
  int   code;
  char *message;
} AppError;

void app_error_set(AppError **err, int domain, int code,
                   const char *fmt, ...) {
  if (!err) return;
  *err = calloc(1, sizeof(**err));
  (*err)->domain = domain; (*err)->code = code;
  va_list ap; va_start(ap, fmt);
  vasprintf(&(*err)->message, fmt, ap);
  va_end(ap);
}

void app_error_free(AppError *err) {
  if (err) { free(err->message); free(err); }
}

/* 对应 g_autofree 的 cleanup wrapper */
static inline void cleanup_apperror(AppError **e) { app_error_free(*e); }
#define autoerr  __attribute__((cleanup(cleanup_apperror)))

/* 使用示例 */
AppError *read_config(const char *path) {
  return NULL; /* 成功时返回 NULL */
}

void init(void) {
  autoerr AppError *err = NULL;
  app_error_set(&err, 1, 404, "File not found: %s", "/etc/app.conf");
  if (err) fprintf(stderr, "Error[%d:%d]: %s\n",
                   err->domain, err->code, err->message);
  /* err 在函数结束时自动释放 */
}

不透明类型封装 — 隐藏实现细节

GLib 广泛使用不透明类型（Opaque Type）模式：结构体内容仅在实现文件中可见，头文件只暴露 typedef 名称与操作函数。GLib 官方文档明确说明诸如 GHashTable、GMainContext、GRegex 等均为"opaque data structure，only accessible via provided functions"。

头文件与实现文件的正确分离

❌ 暴露内部结构（头文件）
/* myobj.h — 错误：暴露实现 */
typedef struct {
  int   state;    /* 用户可随意修改 */
  char *buf;
} MyObject;

MyObject* my_obj_new();
✅ 不透明类型（GLib 风格）
/* myobj.h — 正确：仅前向声明 */
typedef struct _MyObject MyObject;

MyObject* my_obj_new();
void      my_obj_do_work(MyObject*);
void      my_obj_free(MyObject*);

/* myobj.c — 定义藏在实现文件中 */
struct _MyObject { ... };
✅ 优势
  • ABI 稳定：可以自由修改内部布局而不影响调用方的编译
  • 封装性：防止用户绕过接口直接操作内部字段
  • 线程安全更易实现：内部可自由添加锁，外部无感知

销毁回调模式 — GDestroyNotify

GLib 定义了 GDestroyNotify 类型（typedef void (*GDestroyNotify)(gpointer data)），在数据结构、线程本地存储、哈希表等接口中广泛使用，让容器或系统在释放数据时能自动调用用户注册的清理函数，实现了通用的资源销毁钩子。

典型用法场景

GLib APIGDestroyNotify 作用
g_hash_table_new_full(..., key_destroy, value_destroy)哈希表销毁或覆盖条目时自动释放 key/value
g_private_new(GDestroyNotify)线程退出时自动释放线程局部变量
g_source_set_callback(..., GDestroyNotify)GSource 销毁时释放 user_data
g_rc_box_release_full(p, destroy)引用计数归零时调用清理函数
g_object_set_data_full(obj, key, data, destroy)对象数据覆盖或销毁时释放附属数据

在自己代码中实现销毁回调模式

/* 通用销毁回调接口 */
typedef void (*DestroyNotify)(void *data);

/* 带 owner 语义的容器节点 */
typedef struct Node {
  void           *data;
  DestroyNotify   destroy;  /* 可为 NULL（不拥有数据）*/
  struct Node    *next;
} Node;

Node *node_new(void *data, DestroyNotify destroy) {
  Node *n = malloc(sizeof(*n));
  n->data = data; n->destroy = destroy; n->next = NULL;
  return n;
}

void node_free(Node *n) {
  if (n) {
    if (n->destroy) n->destroy(n->data);  /* 自动清理 */
    free(n);
  }
}

/* 使用：哈希表拥有字符串 value */
Node *n = node_new(strdup("hello"), free);
node_free(n);  /* 自动 free("hello") */

/* 使用：不拥有（借用）数据 */
Node *borrowed = node_new(some_ptr, NULL);  /* 不释放 */
💡 设计精髓

通过将 DestroyNotify 与数据同时注册，容器无需知道数据的类型，却能在正确的时机释放它。这是在 C 语言中实现"多态析构"的惯用法，等同于 C++ 中虚析构函数的效果，但没有虚表开销。

所有权语义 — g_steal_pointer

GLib 2.44 引入 g_steal_pointer()，用于在不增加引用计数的情况下显式转移指针所有权，配合自动清理宏使用可避免双重释放（double-free）。

📎 官方文档来源

docs.gtk.org/glib/reference-counting.html 和 docs.gtk.org/glib/memory.html

官方示例：g_steal_pointer 转移所有权

/* 官方文档示例：引用计数对象的所有权转移 */
Person *p = g_rc_box_new(Person);

/* fill_person_details() 在其他地方定义 */
fill_person_details(p);

/* 转移所有权：p 变为 NULL，不增加 ref_count */
add_person_to_database_no_ref(db, g_steal_pointer(&p));
/* 此时 p == NULL，不会在函数结束时被二次释放 */

所有权语义三种模式

在自己代码中实现 steal_pointer

/* 仿 g_steal_pointer 的通用宏 */
#define steal_pointer(pp) ({          \
    __typeof__(*(pp)) _tmp = *(pp);   \
    *(pp) = NULL;                     \
    _tmp;                             \
})

/* 使用示例 */
autofree char *local_buf = malloc(1024);
fill_data(local_buf);
/* 将所有权转移给 result，local_buf 变为 NULL，函数结束不会 double-free */
char *result = steal_pointer(&local_buf);

原子操作与线程安全设计

GLib 通过 gatomicrefcount 和 g_atomic_* 系列函数提供了无锁引用计数和原子基础操作，并封装了跨平台的互斥量（GMutex）、条件变量（GCond）、读写锁（GRWLock）和线程本地存储（GPrivate）。

一次性初始化模式（GOnce）

/* GLib GOnce 单次初始化 — 对应 pthread_once */
static GOnce my_once = G_ONCE_INIT;

static gpointer init_expensive_resource(gpointer arg) {
  /* 只会被调用一次，线程安全 */
  return create_resource();
}

void use_resource(void) {
  Resource *r = g_once(&my_once, init_expensive_resource, NULL);
  do_work(r);
}

/* ── 无需 GLib：用 C11 原子量实现同等效果 ── */
#include <stdatomic.h>
static _Atomic int  once_flag = 0;
static Resource    *global_res = NULL;

void use_resource_c11(void) {
  int expected = 0;
  if (atomic_compare_exchange_strong(&once_flag, &expected, 1)) {
    global_res = create_resource();
    atomic_store(&once_flag, 2);
  }
  while (atomic_load(&once_flag) != 2) ; /* 自旋等待 */
  do_work(global_res);
}

事件循环模式 — GMainLoop

GLib 的 GMainLoop/GMainContext/GSource 体系是一套完整的事件驱动框架。官方文档（docs.gtk.org/glib/main-loop.html）描述其核心为：事件源（GSource）被注册到上下文（GMainContext），主循环（GMainLoop）驱动调度。

事件循环状态机（官方文档描述）

主循环使用示例

/* GLib 主循环基本结构 */
#include <glib.h>

gboolean timer_callback(gpointer user_data) {
  printf("Timer fired!\n");
  return G_SOURCE_CONTINUE;  /* TRUE = 保留定时器 */
}

int main(void) {
  g_autoptr(GMainLoop) loop = g_main_loop_new(NULL, FALSE);

  /* 每 1 秒触发一次 */
  g_timeout_add_seconds(1, timer_callback, NULL);

  g_main_loop_run(loop);  /* 阻塞，直到 g_main_loop_quit() */
  return 0;
  /* loop 在此自动 g_main_loop_unref */
}

在自己代码中实现简易事件循环

/* 无需 GLib 的轻量事件循环骨架 */
#include <poll.h>
#include <stdlib.h>

typedef int (*EventCallback)(int fd, short events, void *udata);

typedef struct {
  struct pollfd *fds;
  EventCallback *cbs;
  void          **udatas;
  int             count;
  int             running;
} EventLoop;

void event_loop_run(EventLoop *loop) {
  loop->running = 1;
  while (loop->running) {
    int n = poll(loop->fds, loop->count, -1);
    if (n <= 0) continue;
    for (int i = 0; i < loop->count; i++) {
      if (loop->fds[i].revents) {
        int keep = loop->cbs[i](loop->fds[i].fd,
                                    loop->fds[i].revents,
                                    loop->udatas[i]);
        if (!keep) event_loop_remove(loop, i);
      }
    }
  }
}

不透明类型 极低（设计规范） ABI 稳定，封装性好 所有公共 API 设计
结构化错误（GError 模式） 低（定义结构体+规则） 错误上下文丰富，易调试 系统/网络/文件 IO 层
销毁回调（DestroyNotify） 低（函数指针） 通用容器析构，避免类型耦合 通用容器/哈希表/队列
引用计数 中（实现 RcHeader） 共享数据生命周期管理 多处共享的复杂对象
所有权语义（steal_pointer） 低（一个宏） 防止双重释放，明确转移语义 配合 RAII 使用
事件循环 高（需完整实现） 高并发非阻塞 IO 网络服务器/守护进程
原子引用计数 中（C11 _Atomic） 多线程安全的引用计数 多线程共享数据
💡 最小化移植策略

第一步：引入 RAII 宏 + 不透明类型规范 → 立即改善代码安全性，零成本

第二步：引入结构化错误 + 销毁回调 → 提升 API 可用性与错误诊断能力

第三步：按需引入引用计数 + 原子操作 → 解决共享所有权与线程安全

第四步：若需异步 IO，再实现事件循环 → 或直接使用 libuv/libev 等成熟库
